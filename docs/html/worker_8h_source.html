19template<typename T>
20concept worker_strategy = std::copyable<T> && requires(const T t)
21{
22 t.defer_at(time_point{}, std::declval<void(*)()>());
23};
24
25template<worker_strategy Strategy>
26class worker
27{
28public:
29 template<typename ...Args>
30 requires (!rpp::constraint::variadic_is_same_type<worker<Strategy>, Args...>)
31 worker(Args&& ...args) : m_strategy{std::forward<Args>(args)...} {}
32
33 worker(const worker&) = default;
34 worker(worker&&) noexcept= default;

36 void schedule(constraint::schedulable_fn auto&& fn) const
37 {
38 schedule(std::chrono::high_resolution_clock::now(), std::forward<decltype(fn)>(fn));
39 }

41 void schedule(time_point time_point, constraint::schedulable_fn auto&& fn) const
42 {
43 m_strategy.defer_at(time_point, scheduler_wrapper<std::decay_t<decltype(fn)>>{m_strategy, time_point, std::forward<decltype(fn)>(fn)});
44 }
45
46private:
47 template<constraint::schedulable_fn Fn>
48 struct scheduler_wrapper
49 {
50 scheduler_wrapper(const Strategy& strategy, time_point time_point, const Fn& fn)
51 : m_strategy{strategy}
52 , m_time_point{time_point}
53 , m_fn{fn} {}
54
55 scheduler_wrapper(const Strategy& strategy, time_point time_point, Fn&& fn)
56 : m_strategy{strategy}
57 , m_time_point{time_point}
58 , m_fn{std::move(fn)} {}
59
60 scheduler_wrapper(const scheduler_wrapper&) = default; // LCOV_EXCL_LINE
61 scheduler_wrapper(scheduler_wrapper&&) noexcept(std::is_nothrow_move_constructible_v<Fn>) = default;
62
63 void operator()()
64 {
65 if (auto duration = m_fn())
66 {
67 m_time_point += duration.value();
68 auto time_to_schedule = m_time_point;
69 m_strategy.defer_at(time_to_schedule, std::move(*this));
70 }
71 }
72
73 Strategy m_strategy;
74 time_point m_time_point;
75 Fn m_fn{};
76 };
77private:
78 Strategy m_strategy;
79};
