41void iterate(auto&& iterable,
42 const schedulers::constraint::scheduler auto& scheduler,
43 constraint::subscriber auto&& subscriber)
44{
45 if constexpr (constraint::decayed_same_as<decltype(scheduler), schedulers::immediate>)
46 {
47 for (const auto& v : extract_iterable_from_packed(iterable))
48 {
49 if (subscriber.is_subscribed())
50 subscriber.on_next(utils::as_const(v));
51 else
52 return;
53 }
54 subscriber.on_completed();
55 }
56 else
57 {
58 auto worker = scheduler.create_worker(subscriber.get_subscription());
59 worker.schedule([iterable=std::forward<decltype(iterable)>(iterable),
60 subscriber=std::forward<decltype(subscriber)>(subscriber),
61 index = size_t{0}]() mutable-> schedulers::optional_duration
62 {
63 try
64 {
65 const auto& extracted_iterable = extract_iterable_from_packed(iterable);
66 const auto end = std::cend(extracted_iterable);
67 auto itr = std::cbegin(extracted_iterable);
68
69 std::ranges::advance(itr, static_cast<int64_t>(index), end);
70
71 if (itr != end)
72 {
73 subscriber.on_next(utils::as_const(*itr));
74 if (std::next(itr) != end) // it was not last
75 {
76 ++index;
77 return schedulers::duration{}; // re-schedule this
78 }
79 }
80
81 subscriber.on_completed();
82 }
83 catch(...)
84 {
85 subscriber.on_error(std::current_exception());
86 }
87 return std::nullopt;
88
89 });
90 }
91}

93template<memory_model memory_model, std::ranges::range Container, typename ...Ts>
94auto pack_to_container(Ts&& ...items)
95{
96 if constexpr (memory_model == memory_model::use_stack)
97 return Container{std::forward<Ts>(items)...};
98 else
99 // raw new call to avoid extra copy/moves for items
100 return std::shared_ptr<Container>(new Container{std::forward<Ts>(items)...});
101}

103template<memory_model memory_model, constraint::decayed_type T, typename ...Ts>
104auto pack_variadic(Ts&& ...items)
105{
106 return pack_to_container<memory_model, std::array<T, sizeof...(Ts)>>(std::forward<Ts>(items)...);
107}

240template<memory_model memory_model /* = memory_model::use_stack */>
241auto from_callable(std::invocable<> auto&& callable) requires rpp::details::is_header_included<rpp::details::from_tag, decltype(callable)>
242{
243 auto obs = just<memory_model>(std::forward<decltype(callable)>(callable));
244
245 if constexpr (std::same_as<utils::decayed_invoke_result_t<decltype(callable)>, void>)
246 return std::move(obs).map([](const auto& fn) { fn(); return utils::none{};});
247 else
248 return std::move(obs).map([](const auto& fn) { return fn(); });
249}
