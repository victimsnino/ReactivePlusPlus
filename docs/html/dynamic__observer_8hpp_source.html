31template<constraint::decayed_type T>
32class dynamic_observer final : public interface_observer<T>
33{
34public:
35 template<constraint::on_next_fn<T> OnNext = utils::empty_function_t<T>,
36 constraint::on_error_fn OnError = utils::rethrow_error_t,
37 constraint::on_completed_fn OnCompleted = utils::empty_function_t<>>
38 dynamic_observer(OnNext&& on_next = {}, OnError&& on_error = {}, OnCompleted&& on_completed = {})
39 : m_state{make_shared_state(std::forward<OnNext>(on_next),
40 std::forward<OnError>(on_error),
41 std::forward<OnCompleted>(on_completed))} { }
42
43 dynamic_observer(constraint::on_next_fn<T> auto&& on_next, constraint::on_completed_fn auto&& on_completed)
44 : dynamic_observer{std::forward<decltype(on_next)>(on_next),
45 utils::rethrow_error_t{},
46 std::forward<decltype(on_completed)>(on_completed)} {}
47
48 template<constraint::observer_of_type<T> TObserver>
49 requires (!std::is_same_v<std::decay_t<TObserver>, dynamic_observer<T>>)
50 dynamic_observer(TObserver&& obs)
51 : m_state{std::make_shared<std::decay_t<TObserver>>(std::forward<TObserver>(obs))} {}
52
53 dynamic_observer(const dynamic_observer<T>&) = default;
54 dynamic_observer(dynamic_observer<T>&&) noexcept = default;
55
56 void on_next(const T& v) const override { m_state->on_next(v); }
57 void on_next(T&& v) const override { m_state->on_next(std::move(v)); }
58 void on_error(const std::exception_ptr& err) const override { m_state->on_error(err); }
59 void on_completed() const override { m_state->on_completed(); }
60
64 const dynamic_observer<T>& as_dynamic() const { return *this; }
65
66private:
67 template<typename ...Args>
68 static auto make_shared_state(Args&&...args)
69 {
70 return std::make_shared<specific_observer<T, std::decay_t<Args>...>>(std::forward<Args>(args)...);
71 }
72
73 std::shared_ptr<interface_observer<T>> m_state;
74};
75
76template<constraint::observer TObserver>
77dynamic_observer(TObserver) -> dynamic_observer<utils::extract_observer_type_t<TObserver>>;
78
79template<typename OnNext, typename ...Args>
80dynamic_observer(OnNext, Args...) -> dynamic_observer<utils::decayed_function_argument_t<OnNext>>;
81} // namespace rpp
