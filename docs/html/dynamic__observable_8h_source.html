42template<constraint::decayed_type Type>
43class dynamic_observable final : public interface_observable<Type, dynamic_observable<Type>>
44{
45public:
46 template<constraint::on_subscribe_fn<Type> OnSubscribeFn>
47 dynamic_observable(OnSubscribeFn&& on_subscribe)
48 : m_observable{std::make_shared<specific_observable<Type, std::decay_t<OnSubscribeFn>>>(on_subscribe)} {}
49
50 template<constraint::observable TObs>
51 requires (!std::is_same_v<std::decay_t<TObs>, dynamic_observable<Type>>)
52 dynamic_observable(TObs&& observable)
53 : m_observable{ std::make_shared<std::decay_t<TObs>>(std::forward<TObs>(observable)) } {}
54
55 dynamic_observable(const dynamic_observable<Type>&) = default;
56 dynamic_observable(dynamic_observable<Type>&&) noexcept = default;
57
58 subscription subscribe(const dynamic_subscriber<Type>& subscriber) const noexcept override
59 {
60 return m_observable->subscribe(subscriber);
61 }
62
68 template<typename ...Args>
69 requires std::is_constructible_v<dynamic_subscriber<Type>, std::decay_t<Args>...>
70 subscription subscribe(Args&&...args) const noexcept
71 {
72 return m_observable->subscribe(dynamic_subscriber<Type>{std::forward<Args>(args)...});
73 }
74
75 dynamic_observable<Type> as_dynamic() const { return *this; }
76
77private:
78 std::shared_ptr<virtual_observable<Type>> m_observable{};
79};
