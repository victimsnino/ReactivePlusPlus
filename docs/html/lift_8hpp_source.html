31 auto make_lift_action_by_callbacks(OnNext&& on_next, OnError&& on_error, OnCompleted&& on_completed)
32 {
33  return [on_next = std::forward<OnNext>(on_next),
34  on_error = std::forward<OnError>(on_error),
35  on_completed = std::forward<OnCompleted>(on_completed)]<constraint::subscriber TSub>(TSub&& subscriber)
36  {

38  return create_subscriber_with_state<Type>(std::move(subscription),

46 template<constraint::decayed_type NewType, lift_fn<NewType> OperatorFn, typename TObs>
47 auto lift_impl(OperatorFn&& op, TObs&& _this)
48 {
49  auto action = [_this = std::forward<TObs>(_this), op = std::forward<OperatorFn>(op)]<constraint::subscriber_of_type<NewType> TSub>(TSub&& subscriber)
50  {
51  _this.subscribe(op(std::forward<TSub>(subscriber)));
52  };
53 
54  return specific_observable<NewType, decltype(action)>(std::move(action));
