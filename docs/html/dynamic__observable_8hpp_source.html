17namespace rpp::details
18{
19template<constraint::decayed_type Type>
20class dynamic_observable_state
21{
22public:
23 template<constraint::observable_of_type<Type> TObs>
24 dynamic_observable_state(TObs&& obs)
25 : m_impl{std::make_shared<dynamic_observable_state_impl<std::decay_t<TObs>>>(std::forward<TObs>(obs))} {}
26
27 template<constraint::on_subscribe_fn<Type> TOnSub>
28 requires (!constraint::decayed_same_as<TOnSub, dynamic_observable_state<Type>>)
29 dynamic_observable_state(TOnSub&& on_sub)
30 : m_impl{ std::make_shared<dynamic_observable_state_impl<specific_observable<Type, std::decay_t<TOnSub>>>>(std::forward<TOnSub>(on_sub)) } {}
31
32 dynamic_observable_state(const dynamic_observable_state<Type>&) = default;
33 dynamic_observable_state(dynamic_observable_state<Type>&&) noexcept = default;
34
35 composite_subscription operator()(const dynamic_subscriber<Type>& subscriber) const
36 {
37 return (*m_impl)(subscriber);
38 }
39private:
40 struct interface_dynamic_observable_state_impl
41 {
42 virtual ~interface_dynamic_observable_state_impl() = default;
43
44 virtual composite_subscription operator()(const dynamic_subscriber<Type>& subscriber) const = 0;
45 };
46
47 template<constraint::observable TObs>
48 class dynamic_observable_state_impl final : public interface_dynamic_observable_state_impl
49 {
50 public:
51 dynamic_observable_state_impl(TObs&& observable)
52 : m_observable{std::move(observable)} {}
53
54 dynamic_observable_state_impl(const TObs& observable)
55 : m_observable{observable} {}
56
57 composite_subscription operator()(const dynamic_subscriber<Type>& subscriber) const override
58 {
59 return m_observable.subscribe(subscriber);
60 }
61
62 private:
63 TObs m_observable{};
64 };
65
66 std::shared_ptr<interface_dynamic_observable_state_impl> m_impl{};
67};
68} // namespace rpp::details

80template<constraint::decayed_type Type>
81class dynamic_observable : public specific_observable<Type, details::dynamic_observable_state<Type>>
82{
83public:
84 using base = specific_observable<Type, details::dynamic_observable_state<Type>>;
85 using base::base;
86
87 dynamic_observable(constraint::on_subscribe_fn<Type> auto&& on_subscribe)
88 : base{std::forward<decltype(on_subscribe)>(on_subscribe)} {}
89
90 template<constraint::observable_of_type<Type> TObs>
91 requires (!std::is_same_v<std::decay_t<TObs>, dynamic_observable<Type>>)
92 dynamic_observable(TObs&& observable)
93 : base{std::forward<TObs>(observable)} {}
94};
