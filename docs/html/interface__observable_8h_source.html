77template<constraint::decayed_type Type, typename OnNext, typename OnError, typename OnCompleted>
78auto make_lift_action_by_callbacks(OnNext&& on_next, OnError&& on_error, OnCompleted&& on_completed)
79{
80 return [on_next = std::forward<OnNext>(on_next),
81 on_error = std::forward<OnError>(on_error),
82 on_completed = std::forward<OnCompleted>(on_completed)](constraint::subscriber auto&& subscriber)
83 {
84 auto subscription = subscriber.get_subscription();
85 return create_subscriber_with_state<Type>(subscription,
86 std::forward<decltype(subscriber)>(subscriber),
87 on_next,
88 on_error,
89 on_completed);
90 };

276 template<constraint::decayed_type NewType, details::lift_fn<NewType> OperatorFn, typename FwdThis>
277 static auto lift_impl(OperatorFn&& op, FwdThis&& _this)
278 {
279 auto action = [new_this = std::forward<FwdThis>(_this), op = std::forward<OperatorFn>(op)](auto&& subscriber)
280 {
281 new_this.subscribe(op(std::forward<decltype(subscriber)>(subscriber)));
282 };
