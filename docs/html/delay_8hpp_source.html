27struct delay_on_next
28{
29 schedulers::duration delay;
30
31 void operator()(auto&& value, const auto& subscriber, const auto& worker) const
32 {
33 worker.schedule(delay,
34 [value = std::forward<decltype(value)>(value), subscriber]()
35 {
36 subscriber.on_next(std::move(value));
37 return schedulers::optional_duration{};
38 });
39 }
40};

45struct delay_on_error
46{
47 void operator()(const std::exception_ptr& err, const auto& subscriber, const auto& worker) const
48 {
49 // on-error must be delivered as soon as possible
50 worker.schedule([err, subscriber]()
51 {
52 subscriber.on_error(err);
53 return schedulers::optional_duration{};
54 });
55 }
56};

61struct delay_on_completed
62{
63 schedulers::duration delay;
64
65 void operator()(const auto& subscriber, const auto& worker) const
66 {
67 worker.schedule(delay,
68 [subscriber]()
69 {
70 subscriber.on_completed();
71 return schedulers::optional_duration{};
72 });
73 }
74};

79template<constraint::decayed_type Type, schedulers::constraint::scheduler TScheduler>
80struct delay_impl
81{
82 RPP_NO_UNIQUE_ADDRESS TScheduler scheduler;
83 schedulers::duration delay;
84
85 template<constraint::subscriber_of_type<Type> TSub>
86 auto operator()(TSub&& subscriber) const
87 {
88 auto worker = scheduler.create_worker(subscriber.get_subscription());
89
90 auto subscription = subscriber.get_subscription().make_child();
91 return create_subscriber_with_state<Type>(std::move(subscription),
92 delay_on_next{delay},
93 delay_on_error{},
94 delay_on_completed{delay},
95 std::forward<TSub>(subscriber),
96 std::move(worker));
97 }
98};
