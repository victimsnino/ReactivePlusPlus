46void iterate(const auto& iterable,
47 const schedulers::constraint::scheduler auto& scheduler,
48 const constraint::subscriber auto& subscriber)
49{
50 constexpr auto extract_iterable = []<typename T>(const T& v) -> const auto&
51 {
52 if constexpr (std::ranges::range<T>)
53 return v;
54 else
55 return *v;
56 };
57
58 if constexpr (constraint::decayed_same_as<decltype(scheduler), schedulers::immediate>)
59 {
60 for (const auto& v : extract_iterable(iterable))
61 subscriber.on_next(utilities::as_const(v));
62 subscriber.on_completed();
63 }
64 else
65 {
66 auto worker = scheduler.create_worker(subscriber.get_subscription());
67 worker.schedule([=, index = size_t{0}]() mutable-> schedulers::optional_duration
68 {
69 const auto end = std::cend(extract_iterable(iterable));
70 auto itr = std::cbegin(extract_iterable(iterable));
71
72 std::ranges::advance(itr, index, end);
73
74 if (itr != end)
75 {
76 subscriber.on_next(utilities::as_const(*itr));
77 if (std::next(itr) != end) // it was not last
78 {
79 ++index;
80 return schedulers::duration{}; // re-schedule this
81 }
82 }
83
84 subscriber.on_completed();
85 return {};
86 });
87 }
88}

90template<memory_model memory_model, std::ranges::range Container, typename ...Ts>
91auto pack_to_container(Ts&& ...items)
92{
93 if constexpr (memory_model == memory_model::use_stack)
94 return Container{std::forward<Ts>(items)...};
95 else
96 // raw new call to avoid extra copy/moves for items
97 return std::shared_ptr<Container>(new Container{std::forward<Ts>(items)...});
98}

112auto from(std::ranges::range auto&& iterable, const TScheduler& scheduler)
113{
114 using Container = std::decay_t<decltype(iterable)>;
115 using Type = std::ranges::range_value_t<Container>;
116 return create<Type>([=, items = details::pack_to_container<memory_model, Container>(std::forward<decltype(iterable)>(iterable))](const constraint::subscriber auto& subscriber)

118 details::iterate(items, scheduler, subscriber);
