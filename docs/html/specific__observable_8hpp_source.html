30template<constraint::decayed_type Type, constraint::on_subscribe_fn<Type> OnSubscribeFn>
31class specific_observable : public interface_observable<Type, specific_observable<Type, OnSubscribeFn>>
32{
33public:
34 specific_observable(OnSubscribeFn&& on_subscribe)
35 : m_state{ std::move(on_subscribe) } {}
36
37 specific_observable(const OnSubscribeFn& on_subscribe)
38 : m_state{ on_subscribe } {}
39
40 specific_observable(const specific_observable<Type, OnSubscribeFn>&) = default;
41 specific_observable(specific_observable<Type, OnSubscribeFn>&&) noexcept = default;
42
46 [[nodiscard]] auto as_dynamic() const & { return rpp::dynamic_observable<Type>{*this}; }
47 [[nodiscard]] auto as_dynamic() && { return rpp::dynamic_observable<Type>{std::move(*this)}; }
48
54 template<constraint::subscriber_of_type<Type> TSub>
55 composite_subscription subscribe(const TSub& subscriber) const
56 {
57 return subscribe_impl(subscriber);
58 }
59
65 template<constraint::observer_of_type<Type> TObserver>
66 composite_subscription subscribe(TObserver&& observer) const
67 {
68 return subscribe_impl<std::decay_t<TObserver>>(std::forward<TObserver>(observer));
69 }
70
76 template<constraint::observer_of_type<Type> TObserver>
77 composite_subscription subscribe(constraint::decayed_same_as<composite_subscription> auto&& sub, TObserver&& observer) const
78 {
79 return subscribe_impl(specific_subscriber<Type, std::decay_t<TObserver>>{std::forward<decltype(sub)>(sub), std::forward<TObserver>(observer)});
80 }
81
87 template<typename ...Args>
88 requires constraint::specific_subscriber_constructible<Type, std::decay_t<Args>...>
89 composite_subscription subscribe(Args&&...args) const
90 {
91 return subscribe_impl(rpp::make_specific_subscriber<Type>(std::forward<Args>(args)...));
92 }
93
94private:
95 template<constraint::observer_of_type<Type> Obs>
96 composite_subscription subscribe_impl(const specific_subscriber<Type, Obs>& subscriber) const
97 {
98 try
99 {
100 m_state(subscriber);
101 }
102 catch (const std::runtime_error& exc)
103 {
104 if (subscriber.is_subscribed())
105 subscriber.on_error(std::make_exception_ptr(exc));
106 else
107 throw;
108 }
109 return subscriber.get_subscription();
110 }
111private:
112 [[no_unique_address]] OnSubscribeFn m_state;
113};
114
115template<typename OnSub>
116specific_observable(OnSub on_subscribe) -> specific_observable<utils::extract_subscriber_type_t<utils::function_argument_t<OnSub>>, OnSub>;
117} // namespace rpp
