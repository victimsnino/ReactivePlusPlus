40public:

51 subscription_base add(const subscription_base& sub) const
52 {
53 static_cast<state&>(get_state()).add(sub);
54 return sub;
55 }

60 subscription_base add(const callback_subscription& sub) const
61 {
62 return add(static_cast<const subscription_base&>(sub));

65private:
66 class state final : public details::subscription_state
67 {
68 public:
69 state(std::vector<subscription_base>&& deps)
70 : m_deps{std::move(deps)} {}
71
72 void add(const subscription_base& sub)
73 {
74 while (true)
75 {
76 DepsState expected{DepsState::None};
77 if (m_state.compare_exchange_strong(expected, DepsState::Add))
78 {
79 m_deps.push_back(sub);
80
81 m_state.store(DepsState::None);
82 return;
83 }
84
85 if (expected == DepsState::Unsubscribed)
86 {
87 sub.unsubscribe();
88 return;
89 }
90 }
91 }
92
93 private:
94 void on_unsubscribe() override
95 {
96 while (true)
97 {
98 DepsState expected{DepsState::None};
99 if (m_state.compare_exchange_strong(expected, DepsState::Unsubscribed))
100 {
101 std::ranges::for_each(m_deps, &subscription_base::unsubscribe);
102 m_deps.clear();
103 return;
104 }
105 }
106 }
107
108 private:
109 enum class DepsState : uint8_t
110 {
111 None, //< default state
112 Add, //< set it during adding new element into deps. After success -> FallBack to None
113 Unsubscribed //< permanent state after unsubscribe
114 };
115
116 std::atomic<DepsState> m_state{DepsState::None};
117 std::vector<subscription_base> m_deps{};
118 };
119};
