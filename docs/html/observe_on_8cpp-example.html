observe_on.cpp
#include <rpp/rpp.hpp>
#include <iostream>
int main()
{
std::cout << std::this_thread::get_id() << std::endl;
rpp::source::just(10, 15, 20)
.observe_on(rpp::schedulers::new_thread{})
.as_blocking()
.subscribe([](int v) { std::cout << "[" << std::this_thread::get_id() << "] : " << v << "\n"; });
// Template for output:
// TH1
// [TH2]: 10
// [TH2]: 15
// [TH2]: 20
return 0;
}
