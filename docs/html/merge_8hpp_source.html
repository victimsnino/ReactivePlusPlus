47template<constraint::decayed_type Type>
48auto create_proxy_subscriber(constraint::subscriber auto&& subscriber,
49 const std::shared_ptr<state_t>& state,
50 auto&& on_next,
51 auto&& on_error,
52 auto&& on_completed)
53{
54 ++(state->count_of_on_completed);
55
56 auto subscription = subscriber.get_subscription();
57 return create_subscriber_with_state<Type>(subscription.make_child(),
58 std::forward<decltype(subscriber)>(subscriber),
59 std::forward<decltype(on_next)>(on_next),
60 std::forward<decltype(on_error)>(on_error),
61 std::forward<decltype(on_completed)>(on_completed));
62}

64template<constraint::decayed_type Type, typename SpecificObservable>
65auto member_overload<Type, SpecificObservable, merge_tag>::merge_impl()
66{
67 using ValueType = utils::extract_observable_type_t<Type>;
68
69 return []<constraint::subscriber_of_type<ValueType> TSub>(TSub&& subscriber)
70 {
71 auto state = std::make_shared<state_t>();
72
73 auto wrap_under_guard = [state](const auto& callable)
74 {
75 return [state, callable](auto&&...args)
76 {
77 std::lock_guard lock{ state->mutex };
78 callable(std::forward<decltype(args)>(args)...);
79 };
80 };
81
82 auto on_completed = [=](const constraint::subscriber auto& sub)
83 {
84 if (--(state->count_of_on_completed) == 0)
85 sub.on_completed();
86 };
87
88 auto on_new_observable = [=]<constraint::observable TObs>(TObs&& new_observable,
89 const constraint::subscriber auto& sub)
90 {
91 std::forward<TObs>(new_observable).subscribe(create_proxy_subscriber<ValueType>(sub,
92 state,
93 wrap_under_guard(forwarding_on_next{}),
94 wrap_under_guard(forwarding_on_error{}),
95 on_completed));
96 };
97
98 return create_proxy_subscriber<Type>(std::forward<TSub>(subscriber),
99 state,
100 std::move(on_new_observable),
101 wrap_under_guard(forwarding_on_error{}),
102 on_completed);
103 };
104}
