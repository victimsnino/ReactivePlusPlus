33 template<constraint::decayed_type Type>
34 auto merge_impl()
35 {
36  using ValueType = utils::extract_observable_type_t<Type>;
37 
38  return []<constraint::subscriber_of_type<ValueType> TSub>(TSub&& subscriber)
39  {
40  auto state = std::make_shared<merge_state_t>();
41  auto count_of_on_completed = std::shared_ptr<std::atomic_size_t>{ state, &state->count_of_on_completed};
42 
43  auto wrap_under_guard = [state](const auto& callable)
44  {
45  return [state, callable](auto&&...args)
46  {
47  std::lock_guard lock{ state->mutex };
48  callable(std::forward<decltype(args)>(args)...);
49  };
50  };
51 
52  auto on_completed = [=](const constraint::subscriber auto& sub)
53  {
54  if (--(*count_of_on_completed) == 0)
55  sub.on_completed();
56  };
57 
58  auto on_new_observable = [=]<constraint::observable TObs>(TObs&& new_observable,
59  const constraint::subscriber auto& sub)
60  {
61  std::forward<TObs>(new_observable).subscribe(combining::create_proxy_subscriber<ValueType>(sub,
62  count_of_on_completed,
63  wrap_under_guard(forwarding_on_next{}),
64  wrap_under_guard(forwarding_on_error{}),
65  on_completed));
66  };
67 
68  return combining::create_proxy_subscriber<Type>(std::forward<TSub>(subscriber),
69  count_of_on_completed,
70  std::move(on_new_observable),
71  wrap_under_guard(forwarding_on_error{}),
72  on_completed);
73  };
74 }
