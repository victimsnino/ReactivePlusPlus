25 template<rpp::constraint::observer TObserver>
26 class merge_state final
27 {
28 public:
29 merge_state(TObserver&& observer)
30 : m_observer(std::move(observer))
31 {
32 get_observer_under_lock()->set_upstream(m_disposable);
33 }
34
35 // just need atomicity, not guarding anything
36 void increment_on_completed() { m_on_completed_needed.fetch_add(1, std::memory_order::seq_cst); }
37
38 // just need atomicity, not guarding anything
39 bool decrement_on_completed() { return m_on_completed_needed.fetch_sub(1, std::memory_order::seq_cst) == 1; }
40
41 rpp::utils::pointer_under_lock<TObserver> get_observer_under_lock() { return m_observer; }
42
43 const rpp::composite_disposable_wrapper& get_disposable() const { return m_disposable; }
44
45 private:
46 rpp::utils::value_with_mutex<TObserver> m_observer{};
47 rpp::composite_disposable_wrapper m_disposable = composite_disposable_wrapper::make();
48 std::atomic_size_t m_on_completed_needed{1};
49 };

51 template<rpp::constraint::observer TObserver>
52 struct merge_observer_base_strategy
53 {
54 merge_observer_base_strategy(std::shared_ptr<merge_state<TObserver>>&& state)
55 : m_state{std::move(state)}
56 {
57 }
58
59 merge_observer_base_strategy(const std::shared_ptr<merge_state<TObserver>>& state)
60 : m_state{state}
61 {
62 }
63
64 void set_upstream(const rpp::disposable_wrapper& d) const
65 {
66 m_state->get_disposable().add(d);
67 m_disposables.push_back(d);
68 }
69
70 bool is_disposed() const
71 {
72 return m_state->get_disposable().is_disposed();
73 }
74
75 void on_error(const std::exception_ptr& err) const
76 {
77 m_state->get_observer_under_lock()->on_error(err);
78 }
79
80 void on_completed() const
81 {
82 if (m_state->decrement_on_completed())
83 {
84 m_state->get_observer_under_lock()->on_completed();
85 }
86 else
87 {
88 for (const auto& v : m_disposables)
89 {
90 m_state->get_disposable().remove(v);
91 }
92 }
93 }
94
95 protected:
96 std::shared_ptr<merge_state<TObserver>> m_state;
97 mutable std::vector<rpp::disposable_wrapper> m_disposables{};
98 };

166 template<rpp::constraint::observer Observer, typename... Strategies>
167 void subscribe(Observer&& observer, const rpp::details::observables::chain<Strategies...>& observable_strategy) const
168 {
169 merge_observer_strategy<std::decay_t<Observer>> strategy{std::forward<Observer>(observer)};
170
171 // Need to take ownership over current_thread in case of inner-observables also using it
172 auto drain_on_exit = rpp::schedulers::current_thread::own_queue_and_drain_finally_if_not_owned();
173
174 strategy.on_next(observable_strategy);
175 observables.apply(&apply<std::decay_t<Observer>>, strategy);
176 strategy.on_completed();
177 }
178
179 private:
180 template<rpp::constraint::observer Observer>
181 static void apply(const merge_observer_strategy<Observer>& strategy, const TObservables&... observables)
182 {
183 (strategy.on_next(observables), ...);
184 }
