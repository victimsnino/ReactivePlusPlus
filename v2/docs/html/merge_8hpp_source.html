26template<rpp::constraint::observer TObserver>
27class merge_disposable final : public composite_disposable
28{
29public:
30 merge_disposable(TObserver&& observer)
31 : m_observer(std::move(observer))
32 {
33 }
34
35 // just need atomicity, not guarding anything
36 void increment_on_completed() { m_on_completed_needed.fetch_add(1, std::memory_order::seq_cst); }
37
38 // just need atomicity, not guarding anything
39 bool decrement_on_completed() { return m_on_completed_needed.fetch_sub(1, std::memory_order::seq_cst) == 1; }
40
41 pointer_under_lock<TObserver> get_observer_under_lock() { return pointer_under_lock{m_observer}; }
42
43private:
44 value_with_mutex<TObserver> m_observer{};
45 std::atomic_size_t m_on_completed_needed{1};
46};

48template<rpp::constraint::observer TObserver>
49struct merge_observer_base_strategy
50{
51 merge_observer_base_strategy(std::shared_ptr<merge_disposable<TObserver>>&& disposable)
52 : m_disposable{std::move(disposable)}
53 {
54 }
55
56 merge_observer_base_strategy(const std::shared_ptr<merge_disposable<TObserver>>& disposable)
57 : m_disposable{disposable}
58 {
59 }
60
61 void set_upstream(const rpp::disposable_wrapper& d) const
62 {
63 m_disposable->add(d);
64 m_disposables.push_back(d);
65 }
66
67 bool is_disposed() const
68 {
69 return m_disposable->is_disposed();
70 }
71
72 void on_error(const std::exception_ptr& err) const
73 {
74 m_disposable->dispose();
75 m_disposable->get_observer_under_lock()->on_error(err);
76 }
77
78 void on_completed() const
79 {
80 if (m_disposable->decrement_on_completed())
81 {
82 for (const auto& v : m_disposables) {
83 m_disposable->remove(v);
84 }
85 m_disposable->dispose();
86 m_disposable->get_observer_under_lock()->on_completed();
87 }
88 }
89
90protected:
91 std::shared_ptr<merge_disposable<TObserver>> m_disposable;
92 mutable std::vector<rpp::disposable_wrapper> m_disposables{};
93};

107template<rpp::constraint::observer TObserver>
108class merge_observer_strategy final : public merge_observer_base_strategy<TObserver>
109{
110public:
111 explicit merge_observer_strategy(TObserver&& observer)
112 : merge_observer_base_strategy<TObserver>{init_state(std::move(observer))}
113 {
114 }
115
116 template<typename T>
117 void on_next(T&& v) const
118 {
119 merge_observer_base_strategy<TObserver>::m_disposable->increment_on_completed();
120 std::forward<T>(v).subscribe(rpp::observer<rpp::utils::extract_observer_type_t<TObserver>, merge_observer_inner_strategy<TObserver>>{merge_observer_inner_strategy<TObserver>{merge_observer_base_strategy<TObserver>::m_disposable}});
121 }
122
123private:
124 static std::shared_ptr<merge_disposable<TObserver>> init_state(TObserver&& observer)
125 {
126 const auto d = disposable_wrapper_impl<merge_disposable<TObserver>>::make(std::move(observer));
127 auto ptr = d.lock();
128 ptr->get_observer_under_lock()->set_upstream(d.as_weak());
129 return ptr;
130 }

154template<rpp::constraint::observable... TObservables>
155struct merge_with_t
156{
157 RPP_NO_UNIQUE_ADDRESS rpp::utils::tuple<TObservables...> observables{};
158
159 template<rpp::constraint::decayed_type T>
160 requires (std::same_as<T, rpp::utils::extract_observable_type_t<TObservables>> && ...)
161 using result_value = T;
162
163 template<rpp::details::observables::constraint::disposable_strategy Prev>
164 using updated_disposable_strategy = rpp::details::observables::fixed_disposable_strategy_selector<1>;
165
166 template<rpp::constraint::observer Observer, typename... Strategies>
167 void subscribe(Observer&& observer, const observable_chain_strategy<Strategies...>& observable_strategy) const
168 {
169 merge_observer_strategy<std::decay_t<Observer>> strategy{std::forward<Observer>(observer)};
170
171 // Need to take ownership over current_thread in case of inner-observables also using it
172 auto drain_on_exit = rpp::schedulers::current_thread::own_queue_and_drain_finally_if_not_owned();
173
174 strategy.on_next(observable_strategy);
175 observables.apply(&apply<std::decay_t<Observer>>, strategy);
176 strategy.on_completed();
177 }
178
179private:
180 template<rpp::constraint::observer Observer>
181 static void apply(const merge_observer_strategy<Observer>& strategy, const TObservables&... observables)
182 {
183 (strategy.on_next(observables), ...);
184 }
185};
