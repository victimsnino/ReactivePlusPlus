74template<rpp::constraint::observer Observer, typename Worker, rpp::details::disposables::constraint::disposable_container Container, bool ClearOnError>
75struct delay_observer_strategy
76{
77 std::shared_ptr<delay_disposable<Observer, Worker, Container>> disposable{};
78
79 void set_upstream(const rpp::disposable_wrapper& d) const
80 {
81 disposable->add(d);
82 }
83
84 bool is_disposed() const
85 {
86 return disposable->is_disposed();
87 }
88
89 template<typename T>
90 void on_next(T&& v) const
91 {
92 emplace(std::forward<T>(v));
93 }
94
95 void on_error(const std::exception_ptr& err) const noexcept
96 {
97 emplace(err);
98 }
99
100 void on_completed() const noexcept
101 {
102 emplace(rpp::utils::none{});
103 }
104
105private:
106 template<typename TT>
107 void emplace(TT&& value) const
108 {
109 if (const auto delay = emplace_safe(std::forward<TT>(value)))
110 {
111 disposable->worker.schedule(
112 delay.value(),
113 [](const delay_disposable_wrapper<Observer, Worker, Container>& wrapper) { return drain_queue(wrapper.disposable); },
114 delay_disposable_wrapper<Observer, Worker, Container>{disposable});
115 }
116 }
117
118 template<typename TT>
119 std::optional<rpp::schedulers::duration> emplace_safe(TT&& item) const
120 {
121 std::lock_guard lock{disposable->mutex};
122 if constexpr (ClearOnError && rpp::constraint::decayed_same_as<std::exception_ptr, TT>)
123 {
124 disposable->queue = std::queue<emission<rpp::utils::extract_observer_type_t<Observer>>>{};
125 disposable->observer.on_error(std::forward<TT>(item));
126 return std::nullopt;
127 }
128 else
129 {
130 disposable->queue.emplace(std::forward<TT>(item), disposable->worker.now() + disposable->delay);
131 if (!disposable->is_active)
132 {
133 disposable->is_active = true;
134 return disposable->delay;
135 }
136 return std::nullopt;
137 }
138 }
139
140 static schedulers::optional_delay_to drain_queue(const std::shared_ptr<delay_disposable<Observer, Worker, Container>>& disposable)
141 {
142 while (true)
143 {
144 std::unique_lock lock{disposable->mutex};
145 if (disposable->queue.empty())
146 {
147 disposable->is_active = false;
148 return std::nullopt;
149 }
150
151 auto& top = disposable->queue.front();
152 if (top.time_point > disposable->worker.now())
153 return schedulers::optional_delay_to{top.time_point};
154
155 auto item = std::move(top.value);
156 disposable->queue.pop();
157 lock.unlock();
158
159 std::visit(rpp::utils::overloaded{[&](rpp::utils::extract_observer_type_t<Observer>&& v) { disposable->observer.on_next(std::move(v)); },
160 [&](const std::exception_ptr& err) { disposable->observer.on_error(err); },
161 [&](rpp::utils::none) {
162 disposable->observer.on_completed();
163 }},
164 std::move(item));
165 }
166 }
167};

169template<rpp::schedulers::constraint::scheduler Scheduler, bool ClearOnError>
170struct delay_t
171{
172 template<rpp::constraint::decayed_type T>
173 using result_value = T;
174
175 template<rpp::details::observables::constraint::disposable_strategy Prev>
176 using updated_disposable_strategy = rpp::details::observables::fixed_disposable_strategy_selector<1>;
177
178 rpp::schedulers::duration duration;
179 RPP_NO_UNIQUE_ADDRESS Scheduler scheduler;
180
181 template<rpp::constraint::decayed_type Type, rpp::details::observables::constraint::disposable_strategy DisposableStrategy, rpp::constraint::observer Observer>
182 auto lift_with_disposable_strategy(Observer&& observer) const
183 {
184 using worker_t = rpp::schedulers::utils::get_worker_t<Scheduler>;
185 using container = typename DisposableStrategy::template add<worker_t::is_none_disposable ? 0 : 1>::disposable_container;
186
187 auto disposable = std::make_shared<delay_disposable<std::decay_t<Observer>, worker_t, container>>(std::forward<Observer>(observer), scheduler.create_worker(), duration);
188 disposable->observer.set_upstream(rpp::disposable_wrapper::from_weak(disposable));
189 return rpp::observer<Type, delay_observer_strategy<std::decay_t<Observer>, worker_t, container, ClearOnError>>{std::move(disposable)};
190 }
191};
