19template<rpp::constraint::observable... TObservables>
20struct start_with_t
21{
22 rpp::utils::tuple<TObservables...> observables{};
23
24 template<rpp::constraint::observable TObservable>
25 requires rpp::constraint::observables_of_same_type<TObservable, TObservables...>
26 auto operator()(TObservable&& observable) const
27 {
28 return observables.apply(&apply<TObservable>, std::forward<TObservable>(observable));
29 }
30
31private:
32 template<rpp::constraint::observable TObservable>
33 static auto apply(TObservable&& observable, const TObservables& ...observables)
34 {
35 return rpp::source::concat(observables..., std::forward<TObservable>(observable));
36 }
37};
