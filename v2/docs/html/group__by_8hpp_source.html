58template<rpp::constraint::decayed_type T, rpp::constraint::decayed_type KeySelector, rpp::constraint::decayed_type ValueSelector, rpp::constraint::decayed_type KeyComparator>
59struct group_by_observer_strategy
60{
61 using DisposableStrategyToUseWithThis = rpp::details::none_disposable_strategy;
62
63 using TKey = rpp::utils::decayed_invoke_result_t<KeySelector, T>;
64 using Type = rpp::utils::decayed_invoke_result_t<ValueSelector, T>;
65
66 RPP_NO_UNIQUE_ADDRESS KeySelector key_selector;
67 RPP_NO_UNIQUE_ADDRESS ValueSelector value_selector;
68 RPP_NO_UNIQUE_ADDRESS KeyComparator comparator;
69
70 mutable std::map<TKey, subjects::publish_subject<Type>, KeyComparator> key_to_subject{};
71 std::shared_ptr<details::with_auto_dispose<refcount_disposable>> disposable = std::make_shared<details::with_auto_dispose<refcount_disposable>>();
72
73 void on_subscribe(rpp::constraint::observer auto& obs) const
74 {
75 obs.set_upstream(rpp::disposable_wrapper{disposable});
76 }
77
78 void set_upstream(const rpp::constraint::observer auto&, const rpp::disposable_wrapper& d) const
79 {
80 disposable->add(d.get_original());
81 }
82
83 bool is_disposed(const rpp::constraint::observer auto&) const
84 {
85 return disposable->is_disposed();
86 }
87
88 template<rpp::constraint::decayed_same_as<T> TT>
89 void on_next(const rpp::constraint::observer auto& obs, TT&& val) const
90 {
91 const auto subject = deduce_subject(obs, val);
92 if (!subject)
93 return;
94
95 const auto& subject_obs = subject->get_observer();
96 if (!subject_obs.is_disposed())
97 subject_obs.on_next(value_selector(std::forward<TT>(val)));
98 }
99
100 void on_error(const rpp::constraint::observer auto& obs, const std::exception_ptr& err) const
101 {
102 for (const auto& [_, subject] : key_to_subject)
103 subject.get_observer().on_error(err);
104
105 obs.on_error(err);
106 }
107
108 void on_completed(const rpp::constraint::observer auto& obs) const
109 {
110 for (const auto& [_, subject] : key_to_subject)
111 subject.get_observer().on_completed();
112
113 obs.on_completed();
114 }
115
116private:
117 template<rpp::constraint::decayed_same_as<T> TT>
118 const subjects::publish_subject<Type>* deduce_subject(const rpp::constraint::observer auto& obs, const TT& val) const
119 {
120 auto key = key_selector(utils::as_const(val));
121
122 if (obs.is_disposed())
123 {
124 const auto itr = key_to_subject.find(key);
125 return itr == key_to_subject.cend() ? nullptr : &itr->second;
126 }
127
128 auto [itr, inserted] = key_to_subject.try_emplace(key);
129
130 if (inserted)
131 {
132 disposable->add(itr->second.get_disposable().get_original());
133 obs.on_next(rpp::grouped_observable_group_by<TKey, Type>{std::move(key), group_by_observable_strategy<Type>{itr->second, disposable}});
134 }
135
136 return &itr->second;
137 }
138};

140template<rpp::constraint::decayed_type T>
141struct group_by_observable_strategy
142{
143 using ValueType = T;
144
145 rpp::subjects::publish_subject<T> subj;
146 std::weak_ptr<refcount_disposable> disposable;
147
148 template<rpp::constraint::observer_strategy<T> Strategy>
149 void subscribe(observer<T, Strategy>&& obs) const
150 {
151 if (const auto locked = disposable.lock())
152 {
153 obs.set_upstream(rpp::disposable_wrapper::from_weak(locked->add_ref().get_original()));
154 subj.get_observable()
155 .subscribe(rpp::observer<T, operator_strategy_base<T, observer<T, Strategy>, group_by_inner_observer_strategy>>{std::move(obs), disposable});
156 }
157 }
158};
