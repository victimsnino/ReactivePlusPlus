62template<rpp::constraint::decayed_type T, rpp::constraint::observer TObserver, rpp::constraint::decayed_type KeySelector, rpp::constraint::decayed_type ValueSelector, rpp::constraint::decayed_type KeyComparator>
63struct group_by_observer_strategy
64{
65 using preferred_disposable_strategy = rpp::details::observers::none_disposable_strategy;
66
67 using TKey = rpp::utils::decayed_invoke_result_t<KeySelector, T>;
68 using Type = rpp::utils::decayed_invoke_result_t<ValueSelector, T>;
69
70 RPP_NO_UNIQUE_ADDRESS TObserver observer;
71 RPP_NO_UNIQUE_ADDRESS KeySelector key_selector;
72 RPP_NO_UNIQUE_ADDRESS ValueSelector value_selector;
73 RPP_NO_UNIQUE_ADDRESS KeyComparator comparator;
74
75 mutable std::map<TKey, subjects::publish_subject<Type>, KeyComparator> key_to_subject{};
76 std::shared_ptr<refcount_disposable> disposable = std::make_shared<refcount_disposable>();
77
78 RPP_CALL_DURING_CONSTRUCTION(
79 {
80 observer.set_upstream(rpp::disposable_wrapper{disposable});
81 });
82
83 void set_upstream(const rpp::disposable_wrapper& d) const
84 {
85 disposable->add(d);
86 }
87
88 bool is_disposed() const
89 {
90 return disposable->is_disposed_underlying();
91 }
92
93 template<rpp::constraint::decayed_same_as<T> TT>
94 void on_next(TT&& val) const
95 {
96 const auto subject = deduce_subject(observer, val);
97 if (!subject)
98 return;
99
100 const auto& subject_obs = subject->get_observer();
101 if (!subject_obs.is_disposed())
102 subject_obs.on_next(value_selector(std::forward<TT>(val)));
103 }
104
105 void on_error(const std::exception_ptr& err) const
106 {
107 for (const auto& [key, subject] : key_to_subject)
108 subject.get_observer().on_error(err);
109
110 observer.on_error(err);
111 }
112
113 void on_completed() const
114 {
115 for (const auto& [key, subject] : key_to_subject)
116 subject.get_observer().on_completed();
117
118 observer.on_completed();
119 }
120
121private:
122 template<rpp::constraint::decayed_same_as<T> TT>
123 const subjects::publish_subject<Type>* deduce_subject(const rpp::constraint::observer auto& obs, const TT& val) const
124 {
125 auto key = key_selector(utils::as_const(val));
126
127 if (obs.is_disposed())
128 {
129 const auto itr = key_to_subject.find(key);
130 return itr == key_to_subject.cend() ? nullptr : &itr->second;
131 }
132
133 auto [itr, inserted] = key_to_subject.try_emplace(key);
134
135 if (inserted)
136 {
137 disposable->add(rpp::disposable_wrapper::from_weak(itr->second.get_disposable().get_original()));
138 obs.on_next(rpp::grouped_observable_group_by<TKey, Type>{
139 std::move(key),
140 group_by_observable_strategy<Type>{itr->second, disposable}
141 });
142 }
143
144 return &itr->second;
145 }
146};

148template<rpp::constraint::decayed_type T>
149struct group_by_observable_strategy
150{
151 using value_type = T;
152
153 rpp::subjects::publish_subject<T> subj;
154 std::weak_ptr<refcount_disposable> disposable;
155
156 template<rpp::constraint::observer_strategy<T> Strategy>
157 void subscribe(observer<T, Strategy>&& obs) const
158 {
159 if (const auto locked = disposable.lock())
160 {
161 obs.set_upstream(locked->add_ref());
162 subj.get_observable()
163 .subscribe(rpp::observer<T, group_by_inner_observer_strategy<observer<T, Strategy>>>{std::move(obs), rpp::composite_disposable_wrapper::from_weak(disposable)});
164 }
165 }
166};
