34template<constraint::decayed_type Type, constraint::observer_strategy<Type> Strategy>
35class base_observer final

37public:
38 template<typename ...Args>
39 requires constraint::is_constructible_from<Strategy, Args...>
40 explicit base_observer(std::optional<composite_disposable> disposable, Args&& ...args)
41 : m_upstream{std::move(disposable)}
42 , m_strategy{std::forward<Args>(args)...} {}

44 template<typename ...Args>
45 requires (!constraint::variadic_decayed_same_as<base_observer<Type, Strategy>, Args...> && constraint::is_constructible_from<Strategy, Args&&...>)

47 : m_strategy{std::forward<Args>(args)...} {}

57 void set_upstream(const composite_disposable& d)
58 {
59 m_strategy.set_upstream(d);

104 void on_error(const std::exception_ptr& err) const

117 void on_completed() const
118 {
119 if (!is_disposed())

124 }
125
129 dynamic_observer<Type> as_dynamic() &&
130 {
131 return dynamic_observer<Type>{std::move(m_upstream), std::move(m_strategy)};
132 }
133
134 operator dynamic_observer<Type>() &&
135 {
136 return std::move(*this).as_dynamic();
137 }
138
139private:
140 void dispose() const
141 {
142 m_is_disposed = true;
143 if (m_upstream)
144 m_upstream->dispose();
145 }
146
147 bool is_upstream_disposed() const
148 {
149 return m_upstream && m_upstream->is_disposed();
150 }
151
152private:
153 std::optional<composite_disposable> m_upstream{};
154 RPP_NO_UNIQUE_ADDRESS Strategy m_strategy;
155 mutable bool m_is_disposed = is_upstream_disposed();
156};
