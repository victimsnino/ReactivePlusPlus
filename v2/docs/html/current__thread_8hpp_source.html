104 static void drain_queue() noexcept
105 {
106 while (get_queue() && !get_queue()->is_empty())
107 {
108 auto top = get_queue()->pop();
109 if (top->is_disposed())
110 continue;
111
112 details::sleep_until(top->get_timepoint());
113
114 while (true)
115 {
116 if (const auto res = top->make_advanced_call())
117 {
118 if (!top->is_disposed())
119 {
120 if (get_queue()->is_empty())
121 {
122 if (const auto d = std::get_if<delay_from_now>(&res->get()))
123 {
124 std::this_thread::sleep_for(d->value);
125 }
126 else
127 {
128 details::sleep_until(top->handle_advanced_call(res.value()));
129 }
130 continue;
131 }
132
133 get_queue()->emplace(top->handle_advanced_call(res.value()), std::move(top));
134 }
135 }
136 break;
137 }
138 }
139
140 get_queue() = nullptr;
141 }

146 template<rpp::schedulers::constraint::schedulable_handler Handler, typename... Args, constraint::schedulable_fn<Handler, Args...> Fn>
147 static void defer_for(duration duration, Fn&& fn, Handler&& handler, Args&&... args)
148 {
149 if (handler.is_disposed())
150 return;
151
152 if (!get_queue())
153 {
154 details::schedulables_queue<worker_strategy> queue{};
155 get_queue() = &queue;
156
157 const auto timepoint = details::immediate_scheduling_while_condition<worker_strategy>(duration, is_queue_is_empty{queue}, fn, handler, args...);
158 if (!timepoint || handler.is_disposed())
159 return drain_queue();
160
161 get_queue()->emplace(timepoint.value(), std::forward<Fn>(fn), std::forward<Handler>(handler), std::forward<Args>(args)...);
162 return drain_queue();
163 }
164
165 get_queue()->emplace(now() + duration, std::forward<Fn>(fn), std::forward<Handler>(handler), std::forward<Args>(args)...);
166 }
167
168 template<rpp::schedulers::constraint::schedulable_handler Handler, typename... Args, constraint::schedulable_fn<Handler, Args...> Fn>
169 static void defer_to(time_point tp, Fn&& fn, Handler&& handler, Args&&... args)
170 {
171 if (handler.is_disposed())
172 return;
173
174 if (get_queue())
175 {
176 get_queue()->emplace(tp, std::forward<Fn>(fn), std::forward<Handler>(handler), std::forward<Args>(args)...);
177 }
178 else
179 {
180 defer_for(tp - now(), std::forward<Fn>(fn), std::forward<Handler>(handler), std::forward<Args>(args)...);
181 }
182 }
