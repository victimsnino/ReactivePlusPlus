107 static void drain_queue(std::optional<details::schedulables_queue<worker_strategy>>& queue)
108 {
109 while (!queue->is_empty())
110 {
111 auto top = queue->pop();
112 if (top->is_disposed())
113 continue;
114
115 std::optional<time_point> timepoint{top->get_timepoint()};
116 // immediate like scheduling
117 do
118 {
119 if (timepoint && !top->is_disposed())
120 details::sleep_until(top->get_timepoint());
121
122 if (top->is_disposed())
123 timepoint.reset();
124 else
125 timepoint = (*top)();
126
127 } while (queue->is_empty() && timepoint.has_value());
128
129 if (timepoint.has_value())
130 queue->emplace(timepoint.value(), std::move(top));
131 }
132
133 queue.reset();
134 }

136 class worker_strategy
137 {
138 public:
139 template<rpp::schedulers::constraint::schedulable_handler Handler, typename... Args, constraint::schedulable_fn<Handler, Args...> Fn>
140 static void defer_for(duration duration, Fn&& fn, Handler&& handler, Args&&... args)
141 {
142 if (handler.is_disposed())
143 return;
144
145 auto& queue = s_queue;
146 const bool someone_owns_queue = queue.has_value();
147 std::optional<time_point> timepoint{};
148 if (!someone_owns_queue)
149 {
150 queue.emplace();
151
152 timepoint = details::immediate_scheduling_while_condition<worker_strategy>(duration, is_queue_is_empty{queue.value()}, fn, handler, args...);
153 if (!timepoint || handler.is_disposed())
154 return drain_queue(queue);
155 }
156 else
157 {
158 timepoint = now() + duration;
159 }
160
161 queue->emplace(timepoint.value(), std::forward<Fn>(fn), std::forward<Handler>(handler), std::forward<Args>(args)...);
162
163 if (!someone_owns_queue)
164 drain_queue(queue);
165 }
166
167 static constexpr rpp::schedulers::details::none_disposable get_disposable() { return {}; }
168
169 static rpp::schedulers::time_point now() { return details::now(); }
170 };
