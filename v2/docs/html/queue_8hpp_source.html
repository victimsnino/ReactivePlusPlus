116 class specific_schedulable final : public schedulable_base
117 {
118 public:
119 template<rpp::constraint::decayed_same_as<Fn> TFn, typename... TArgs>
120 explicit specific_schedulable(const time_point& time_point, TFn&& in_fn, TArgs&&... in_args)
121 : schedulable_base{time_point}
122 , m_args{std::forward<TArgs>(in_args)...}
123 , m_fn{std::forward<TFn>(in_fn)}
124 {
125 }
126
127 std::optional<time_point> operator()() noexcept override
128 {
129 try
130 {
131 if (const auto res = m_args.apply(m_fn))
132 return get_advanced_call_handler<NowStrategy>()(res.value());
133 }
134 catch (...)
135 {
136 m_args.template get<0>().on_error(std::current_exception());
137 }
138 return std::nullopt;
139 }
140
141 std::optional<advanced_call> make_advanced_call() noexcept override
142 {
143 try
144 {
145 if (const auto res = m_args.apply(m_fn))
146 return advanced_call{res.value()};
147 }
148 catch (...)
149 {
150 m_args.template get<0>().on_error(std::current_exception());
151 }
152 return std::nullopt;
153 }
154
155 time_point handle_advanced_call(const advanced_call& v) noexcept override
156 {
157 return v.visit(get_advanced_call_handler<NowStrategy>());
158 }
159
160 bool is_disposed() const noexcept override { return m_args.template get<0>().is_disposed(); }
161
162 private:
163 RPP_NO_UNIQUE_ADDRESS rpp::utils::tuple<Handler, Args...> m_args;

216 template<rpp::schedulers::constraint::schedulable_handler Handler, typename... Args, constraint::schedulable_fn<Handler, Args...> Fn>
217 void emplace(const time_point& timepoint, Fn&& fn, Handler&& handler, Args&&... args)
218 {
219 using schedulable_type = specific_schedulable<NowStrategy, std::decay_t<Fn>, std::decay_t<Handler>, std::decay_t<Args>...>;
220
221 emplace_impl(std::make_shared<schedulable_type>(timepoint, std::forward<Fn>(fn), std::forward<Handler>(handler), std::forward<Args>(args)...));

224 void emplace(const time_point& timepoint, std::shared_ptr<schedulable_base>&& schedulable)
225 {
226 if (!schedulable)
227 return;
228
229 schedulable->set_timepoint(timepoint);
230 emplace_impl(std::move(schedulable));
231 }

236 {
237 return std::exchange(m_head, m_head->get_next());

245 private:
246 void emplace_impl(std::shared_ptr<schedulable_base>&& schedulable)
247 {
248 // needed in case of new_thread and current_thread shares same queue
249 const auto s = m_shared_data.lock();
250 const rpp::utils::finally_action _{[&] {
251 if (s)
252 s->cv.notify_one();
253 }};
254
255 optional_mutex<std::recursive_mutex> mutex{s ? &s->mutex : nullptr};
256 std::lock_guard lock{mutex};
257
258 if (!m_head || schedulable->get_timepoint() < m_head->get_timepoint())
259 {
260 schedulable->set_next(std::move(m_head));
261 m_head = std::move(schedulable);
262 return;
263 }
264
265 schedulable_base* current = m_head.get();
266 while (const auto& next = current->get_next())
267 {
268 if (schedulable->get_timepoint() < next->get_timepoint())
269 break;
270 current = next.get();
271 }
272
273 current->update_next(std::move(schedulable));
274 }
275
276 private:
277 std::shared_ptr<schedulable_base> m_head{};
278 std::weak_ptr<shared_queue_data> m_shared_data{};
279 };
