28 enum ConcatStage : uint8_t
29 {
30 None = 0,
31 Draining = 1,
32 CompletedWhileDraining = 2,
33 Processing = 3,
34 };

36 template<rpp::constraint::observable TObservable, rpp::constraint::observer TObserver>
37 class concat_state_t final : public rpp::refcount_disposable
38 {
39 public:
40 concat_state_t(TObserver&& observer)
41 : m_observer{std::move(observer)}
42 {
43 }
44
45 pointer_under_lock<TObserver> get_observer() { return pointer_under_lock{m_observer}; }
46 pointer_under_lock<std::queue<TObservable>> get_queue() { return pointer_under_lock{m_queue}; }
47
48 std::atomic<ConcatStage>& stage() { return m_stage; }
49
50 void drain(rpp::composite_disposable_wrapper refcounted)
51 {
52 while (!is_disposed())
53 {
54 const auto observable = get_observable();
55 if (!observable)
56 {
57 stage().store(ConcatStage::None, std::memory_order::relaxed);
58 refcounted.dispose();
59 if (is_disposed())
60 get_observer()->on_completed();
61 return;
62 }
63
64 if (handle_observable_impl(observable.value(), refcounted))
65 return;
66 }
67 }
68
69 void handle_observable(const rpp::constraint::decayed_same_as<TObservable> auto& observable, rpp::composite_disposable_wrapper refcounted)
70 {
71 if (handle_observable_impl(observable, refcounted))
72 return;
73
74 drain(refcounted);
75 }
76
77 private:
78 bool handle_observable_impl(const rpp::constraint::decayed_same_as<TObservable> auto& observable, rpp::composite_disposable_wrapper refcounted)
79 {
80 stage().store(ConcatStage::Draining, std::memory_order::relaxed);
81 refcounted.clear();
82 observable.subscribe(concat_inner_observer_strategy<TObservable, TObserver>{disposable_wrapper_impl<concat_state_t>{wrapper_from_this()}.lock(), std::move(refcounted)});
83
84 ConcatStage current = ConcatStage::Draining;
85 return stage().compare_exchange_strong(current, ConcatStage::Processing, std::memory_order::seq_cst);
86 }
87
88 private:
89 std::optional<TObservable> get_observable()
90 {
91 auto queue = get_queue();
92 if (queue->empty())
93 return std::nullopt;
94 auto observable = queue->front();
95 queue->pop();
96 return observable;
97 }
98
99 private:
100 value_with_mutex<TObserver> m_observer;
101 value_with_mutex<std::queue<TObservable>> m_queue;
102 std::atomic<ConcatStage> m_stage{};
103 };

133 template<rpp::constraint::observable TObservable, rpp::constraint::observer TObserver>
134 struct concat_inner_observer_strategy : public concat_observer_strategy_base<TObservable, TObserver>
135 {
136 using base = concat_observer_strategy_base<TObservable, TObserver>;
137
138 using base::concat_observer_strategy_base;
139
140 template<typename T>
141 void on_next(T&& v) const
142 {
143 base::state->get_observer()->on_next(std::forward<T>(v));
144 }
145
146 void on_completed() const
147 {
148 ConcatStage current{ConcatStage::Draining};
149 if (base::state->stage().compare_exchange_strong(current, ConcatStage::CompletedWhileDraining, std::memory_order::seq_cst))
150 return;
151
152 assert(current == ConcatStage::Processing);
153
154 base::state->drain(base::refcounted);
155 }
156 };

158 template<rpp::constraint::observable TObservable, rpp::constraint::observer TObserver>
159 struct concat_observer_strategy : public concat_observer_strategy_base<TObservable, TObserver>
160 {
161 using base = concat_observer_strategy_base<TObservable, TObserver>;
162 using preferred_disposable_strategy = rpp::details::observers::none_disposable_strategy;
163
164 concat_observer_strategy(TObserver&& observer)
165 : base{init_state(std::move(observer))}
166 {
167 }
168
169 template<typename T>
170 void on_next(T&& v) const
171 {
172 ConcatStage current = ConcatStage::None;
173 if (base::state->stage().compare_exchange_strong(current, ConcatStage::Draining, std::memory_order::seq_cst))
174 base::state->handle_observable(std::forward<T>(v), base::state->add_ref());
175 else
176 base::state->get_queue()->push(std::forward<T>(v));
177 }
178
179 void on_completed() const
180 {
181 base::refcounted.dispose();
182 if (base::state->is_disposed())
183 base::state->get_observer()->on_completed();
184 }
185
186
187 private:
188 static std::shared_ptr<concat_state_t<TObservable, TObserver>> init_state(TObserver&& observer)
189 {
190 const auto d = disposable_wrapper_impl<concat_state_t<TObservable, TObserver>>::make(std::move(observer));
191 auto ptr = d.lock();
192 ptr->get_observer()->set_upstream(d.as_weak());
193 return ptr;
194 }
195 };
