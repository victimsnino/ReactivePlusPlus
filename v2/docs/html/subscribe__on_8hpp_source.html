44 template<rpp::details::observables::constraint::disposable_strategy Prev>
45 using updated_disposable_strategy = typename Prev::template add<rpp::schedulers::utils::get_worker_t<TScheduler>::is_none_disposable ? 0 : 1>;
46
47 RPP_NO_UNIQUE_ADDRESS TScheduler scheduler;
48
49 template<rpp::constraint::observer Observer, typename... Strategies>
50 void subscribe(Observer&& observer, const rpp::details::observables::chain<Strategies...>& observable_strategy) const
51 {
52 const auto worker = scheduler.create_worker();
53 if constexpr (!rpp::schedulers::utils::get_worker_t<TScheduler>::is_none_disposable)
54 {
55 if (auto d = worker.get_disposable(); !d.is_disposed())
56 observer.set_upstream(std::move(d));
57 }
58 worker.schedule(subscribe_on_schedulable<rpp::details::observables::chain<Strategies...>>{observable_strategy}, std::forward<Observer>(observer));
59 }
60 };
