69 static void data_thread(std::shared_ptr<queue_data> state)
70 {
71 current_thread::get_queue() = &state->queue;
72
73 while (true)
74 {
75 std::unique_lock lock{state->mutex};
76 if (state->queue.is_empty() && state->is_stoping)
77 break;
78
79 state->cv.wait(lock, [&] { return !state->queue.is_empty() || state->is_stoping; });
80
81 if (state->queue.is_empty())
82 break;
83
84 if (state->queue.top()->is_disposed())
85 {
86 state->queue.pop();
87 continue;
88 }
89
90 if (details::s_last_now_time < state->queue.top()->get_timepoint())
91 {
92 if (const auto now = worker_strategy::now(); now < state->queue.top()->get_timepoint())
93 {
94 state->cv.wait_for(lock, state->queue.top()->get_timepoint() - now, [&] { return state->queue.top()->is_disposed() || worker_strategy::now() >= state->queue.top()->get_timepoint(); });
95 continue;
96 }
97 }
98
99 auto top = state->queue.pop();
100 state->has_fresh_data.store(!state->queue.is_empty());
101 lock.unlock();
102
103 while (true)
104 {
105 if (const auto res = top->make_advanced_call())
106 {
107 if (!top->is_disposed())
108 {
109 if (res->can_run_immediately() && !state->has_fresh_data.load())
110 continue;
111
112 const auto tp = top->handle_advanced_call(res.value());
113 state->queue.emplace(tp, std::move(top));
114 }
115 }
116 break;
117 }
118 }
119
120 current_thread::get_queue() = nullptr;
121 }
122
123 private:
124 std::shared_ptr<queue_data> m_state = std::make_shared<queue_data>();
125
126 RPP_CALL_DURING_CONSTRUCTION(m_state->queue = details::schedulables_queue<current_thread::worker_strategy>(m_state));
127
128 std::thread m_thread{&data_thread, m_state};
129 };
