54 template<rpp::schedulers::constraint::schedulable_handler Handler, typename... Args, constraint::schedulable_fn<Handler, Args...> Fn>
55 void defer_at(time_point time_point, Fn&& fn, Handler&& handler, Args&&... args)
56 {
57 if (is_disposed())
58 return;
59
60 std::lock_guard lock{m_state->mutex};
61 // guarded by lock
62 if (const auto queue = m_state->queue_ptr.load(std::memory_order::seq_cst))
63 queue->emplace(time_point, std::forward<Fn>(fn), std::forward<Handler>(handler), std::forward<Args>(args)...);
64 }

68 {
69 if (!m_thread.joinable())
70 return;
71
72 // just need atomicity, not guarding anything
73 m_state->is_disposed.store(true, std::memory_order::seq_cst);
74 m_state->cv.notify_all();
75
76 if (m_thread.get_id() != std::this_thread::get_id())
77 m_thread.join();
78 else
79 m_thread.detach();
80 }

89 static void data_thread(std::shared_ptr<state_t> state)
90 {
91 auto& queue = current_thread::s_queue;
92 state->queue_ptr.store(&queue.emplace(state), std::memory_order::seq_cst);
93
94 while (!state->is_disposed.load(std::memory_order::seq_cst))
95 {
96 std::unique_lock lock{state->mutex};
97
98 if (state->is_destroying.load(std::memory_order::seq_cst) && queue->is_empty())
99 break;
100
101 state->cv.wait(lock, [&] { return state->is_disposed.load(std::memory_order::seq_cst) || !queue->is_empty() || state->is_destroying.load(std::memory_order::seq_cst); });
102
103 if (state->is_disposed.load(std::memory_order::seq_cst) || state->is_destroying.load(std::memory_order::seq_cst))
104 break;
105
106 if (queue->top()->is_disposed())
107 {
108 queue->pop();
109 continue;
110 }
111
112 if (details::s_last_now_time < queue->top()->get_timepoint())
113 {
114 if (const auto now = worker_strategy::now(); now < queue->top()->get_timepoint())
115 {
116 state->cv.wait_for(lock, queue->top()->get_timepoint() - now, [&] { return state->is_disposed.load(std::memory_order::seq_cst) || state->is_destroying.load(std::memory_order::seq_cst) || worker_strategy::now() >= queue->top()->get_timepoint(); });
117 continue;
118 }
119 }
120
121 auto top = queue->pop();
122 lock.unlock();
123
124 if (const auto timepoint = (*top)())
125 queue->emplace(timepoint.value(), std::move(top));
126 }
127
128 std::unique_lock lock{state->mutex};
129 state->queue_ptr.store(nullptr, std::memory_order::seq_cst);
130 queue.reset();
131 }
132
133 private:
134 std::shared_ptr<state_t> m_state = std::make_shared<state_t>();
135 std::thread m_thread{&data_thread, m_state};
136 };
