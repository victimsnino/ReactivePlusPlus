34template<rpp::constraint::observer Observer, typename Worker>
35class debounce_disposable final : public rpp::composite_disposable
36 , public std::enable_shared_from_this<debounce_disposable<Observer, Worker>>
37{
38 using T = rpp::utils::extract_observer_type_t<Observer>;
39
40public:
41 debounce_disposable(Observer&& in_observer, Worker&& in_worker, rpp::schedulers::duration period)
42 : m_observer(std::move(in_observer))
43 , m_worker{std::move(in_worker)}
44 , m_period{period}
45 {
46 if (auto d = m_worker.get_disposable(); !d.is_disposed())
47 add(std::move(d));
48 }
49
50 template<typename TT>
51 void emplace_safe(TT&& v)
52 {
53 std::lock_guard lock{m_mutex};
54 m_value_to_be_emitted.emplace(std::forward<TT>(v));
55 const bool need_to_scheduled = !m_time_when_value_should_be_emitted.has_value() || !m_value_to_be_emitted.has_value();
56 m_time_when_value_should_be_emitted = m_worker.now() + m_period;
57 if (need_to_scheduled)
58 {
59 schedule();
60 }
61 }
62
63 std::optional<T> extract_value()
64 {
65 std::lock_guard lock{m_mutex};
66 return std::exchange(m_value_to_be_emitted, std::optional<T>{});
67 }
68
69 pointer_under_lock<Observer> get_observer_under_lock() { return pointer_under_lock{m_observer}; }
70
71private:
72 void schedule()
73 {
74 m_worker.schedule(
75 m_period,
76 [](const debounce_disposable_wrapper<Observer, Worker>& handler) -> schedulers::optional_duration {
77 auto value_or_duration = handler.disposable->extract_value_or_time();
78 if (auto* duration = std::get_if<schedulers::duration>(&value_or_duration))
79 return *duration;
80
81 if (auto* value = std::get_if<T>(&value_or_duration))
82 handler.disposable->get_observer_under_lock()->on_next(std::move(*value));
83
84 return std::nullopt;
85 },
86 debounce_disposable_wrapper<Observer, Worker>{this->shared_from_this()});
87 }
88
89 std::variant<std::monostate, T, schedulers::duration> extract_value_or_time()
90 {
91 std::lock_guard lock{m_mutex};
92 if (!m_time_when_value_should_be_emitted.has_value() || !m_value_to_be_emitted.has_value())
93 return std::monostate{};
94
95 const auto now = m_worker.now();
96 if (m_time_when_value_should_be_emitted > now)
97 return m_time_when_value_should_be_emitted.value() - now;
98
99 m_time_when_value_should_be_emitted.reset();
100 auto v = std::move(m_value_to_be_emitted).value();
101 m_value_to_be_emitted.reset();
102 return v;
103 }
104
105 value_with_mutex<Observer> m_observer;
106 RPP_NO_UNIQUE_ADDRESS Worker m_worker;
107 rpp::schedulers::duration m_period;
108
109 std::mutex m_mutex{};
110 std::optional<schedulers::time_point> m_time_when_value_should_be_emitted{};
111 std::optional<T> m_value_to_be_emitted{};
112};

114template<rpp::constraint::observer Observer, typename Worker>
115struct debounce_observer_strategy
116{
117 std::shared_ptr<debounce_disposable<Observer, Worker>> disposable{};
118
119 void set_upstream(const rpp::disposable_wrapper& d) const
120 {
121 disposable->add(d);
122 }
123
124 bool is_disposed() const
125 {
126 return disposable->is_disposed();
127 }
128
129 template<typename T>
130 void on_next(T&& v) const
131 {
132 disposable->emplace_safe(std::forward<T>(v));
133 }
134
135 void on_error(const std::exception_ptr& err) const noexcept
136 {
137 disposable->dispose();
138 disposable->get_observer_under_lock()->on_error(err);
139 }
140
141 void on_completed() const noexcept
142 {
143 disposable->dispose();
144 const auto value = disposable->extract_value();
145 const auto observer = disposable->get_observer_under_lock();
146 if (value)
147 observer->on_next(std::move(value).value());
148 observer->on_completed();
149 }
150};

152template<rpp::schedulers::constraint::scheduler Scheduler>
153struct debounce_t
154{
155 template<rpp::constraint::decayed_type T>
156 using ResultValue = T;
157
158 rpp::schedulers::duration duration;
159 RPP_NO_UNIQUE_ADDRESS Scheduler scheduler;
160
161 template<rpp::constraint::decayed_type Type, rpp::constraint::observer Observer>
162 auto lift(Observer&& observer) const
163 {
164 using worker_t = decltype(scheduler.create_worker());
165
166 auto disposable = std::make_shared<debounce_disposable<std::decay_t<Observer>, worker_t>>(std::forward<Observer>(observer), scheduler.create_worker(), duration);
167 disposable->get_observer_under_lock()->set_upstream(rpp::disposable_wrapper::from_weak(disposable));
168 return rpp::observer<Type, debounce_observer_strategy<std::decay_t<Observer>, worker_t>>{std::move(disposable)};
169 }
170};
171}
