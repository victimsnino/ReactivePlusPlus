33 template<rpp::constraint::observer Observer, typename Worker>
34 class debounce_state final : public rpp::details::enable_wrapper_from_this<debounce_state<Observer, Worker>>
35 , public rpp::details::base_disposable
36 {
37 using T = rpp::utils::extract_observer_type_t<Observer>;
38
39 public:
40 debounce_state(Observer&& in_observer, Worker&& in_worker, rpp::schedulers::duration period)
41 : m_observer(std::move(in_observer))
42 , m_worker{std::move(in_worker)}
43 , m_period{period}
44 {
45 }
46
47 template<typename TT>
48 void emplace_safe(TT&& v)
49 {
50 std::lock_guard lock{m_mutex};
51 m_value_to_be_emitted.emplace(std::forward<TT>(v));
52 const bool need_to_scheduled = !m_time_when_value_should_be_emitted.has_value();
53 m_time_when_value_should_be_emitted = m_worker.now() + m_period;
54 if (need_to_scheduled)
55 {
56 schedule();
57 }
58 }
59
60 std::optional<T> extract_value()
61 {
62 std::lock_guard lock{m_mutex};
63 return std::exchange(m_value_to_be_emitted, std::optional<T>{});
64 }
65
66 rpp::utils::pointer_under_lock<Observer> get_observer_under_lock() { return m_observer; }
67
68 private:
69 void schedule()
70 {
71 m_worker.schedule(
72 m_time_when_value_should_be_emitted.value(),
73 [](const debounce_state_wrapper<Observer, Worker>& handler) -> schedulers::optional_delay_to {
74 auto value_or_duration = handler.state->extract_value_or_time();
75 if (auto* timepoint = std::get_if<schedulers::time_point>(&value_or_duration))
76 return schedulers::optional_delay_to{*timepoint};
77
78 if (auto* value = std::get_if<T>(&value_or_duration))
79 handler.state->get_observer_under_lock()->on_next(std::move(*value));
80
81 return std::nullopt;
82 },
83 debounce_state_wrapper<Observer, Worker>{this->wrapper_from_this().lock()});
84 }
85
86 std::variant<std::monostate, T, schedulers::time_point> extract_value_or_time()
87 {
88 std::lock_guard lock{m_mutex};
89 if (!m_time_when_value_should_be_emitted.has_value() || !m_value_to_be_emitted.has_value())
90 return std::monostate{};
91
92 if (m_time_when_value_should_be_emitted > m_worker.now())
93 return m_time_when_value_should_be_emitted.value();
94
95 m_time_when_value_should_be_emitted.reset();
96 auto v = std::move(m_value_to_be_emitted).value();
97 m_value_to_be_emitted.reset();
98 return v;
99 }
100
101 rpp::utils::value_with_mutex<Observer> m_observer;
102 RPP_NO_UNIQUE_ADDRESS Worker m_worker;
103 rpp::schedulers::duration m_period;
104
105 std::mutex m_mutex{};
106 std::optional<schedulers::time_point> m_time_when_value_should_be_emitted{};
107 std::optional<T> m_value_to_be_emitted{};
108 };

110 template<rpp::constraint::observer Observer, typename Worker>
111 struct debounce_observer_strategy
112 {
113 using preferred_disposable_strategy = rpp::details::observers::none_disposable_strategy;
114
115 std::shared_ptr<debounce_state<Observer, Worker>> state{};
116
117 void set_upstream(const rpp::disposable_wrapper& d) const
118 {
119 state->get_observer_under_lock()->set_upstream(d);
120 }
121
122 bool is_disposed() const
123 {
124 return state->is_disposed();
125 }
126
127 template<typename T>
128 void on_next(T&& v) const
129 {
130 state->emplace_safe(std::forward<T>(v));
131 }
132
133 void on_error(const std::exception_ptr& err) const noexcept
134 {
135 state->get_observer_under_lock()->on_error(err);
136 }
137
138 void on_completed() const noexcept
139 {
140 const auto observer = state->get_observer_under_lock();
141 if (const auto value = state->extract_value())
142 observer->on_next(std::move(value).value());
143 observer->on_completed();
144 }
145 };

148 struct debounce_t
149 {
150 template<rpp::constraint::decayed_type T>
151 struct operator_traits
152 {
153 using result_type = T;
154 };
155
156 template<rpp::details::observables::constraint::disposable_strategy Prev>
157 using updated_disposable_strategy = typename Prev::template add<1>;
158
159 rpp::schedulers::duration duration;
160 RPP_NO_UNIQUE_ADDRESS Scheduler scheduler;
161
162 template<rpp::constraint::decayed_type Type, rpp::constraint::observer Observer>
163 auto lift(Observer&& observer) const
164 {
165 using worker_t = rpp::schedulers::utils::get_worker_t<Scheduler>;
166
167 auto d = rpp::disposable_wrapper_impl<debounce_state<std::decay_t<Observer>, worker_t>>::make(std::forward<Observer>(observer), scheduler.create_worker(), duration);
168 auto ptr = d.lock();
169 ptr->get_observer_under_lock()->set_upstream(d.as_weak());
170 return rpp::observer<Type, debounce_observer_strategy<std::decay_t<Observer>, worker_t>>{std::move(ptr)};
171 }
172 };
