44 struct zip_observer_strategy final

49 template<typename T>
50 void on_next(T&& v) const
51 {
52 const auto observer = state->get_observer_under_lock();
53 state->get_pendings().template get<I>().push_back(std::forward<T>(v));
54
55 state->get_pendings().apply(&apply_impl<decltype(state)>, state, observer);
56 }
57
58 private:
59 template<typename TDisposable>
60 static void apply_impl(const TDisposable& disposable, const rpp::utils::pointer_under_lock<Observer>& observer, std::deque<Args>&... values)
61 {
62 if ((!values.empty() && ...))
63 {
64 observer->on_next(disposable->get_selector()(std::move(values.front())...));
65 (values.pop_front(), ...);
66 }
67 }
