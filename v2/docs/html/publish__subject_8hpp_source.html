23 template<rpp::constraint::decayed_type Type>
24 class publish_strategy
25 {
26 struct observer_strategy
27 {
28 using preferred_disposable_strategy = rpp::details::observers::none_disposable_strategy;
29
30 std::shared_ptr<subject_state<Type>> state{};
31
32 void set_upstream(const disposable_wrapper& d) const noexcept { state->add(d); }
33
34 bool is_disposed() const noexcept { return state->is_disposed(); }
35
36 void on_next(const Type& v) const { state->on_next(v); }
37
38 void on_error(const std::exception_ptr& err) const { state->on_error(err); }
39
40 void on_completed() const { state->on_completed(); }
41 };
42
43 public:
44 using expected_disposable_strategy = rpp::details::observables::deduce_disposable_strategy_t<subject_state<Type>>;
45
46 auto get_observer() const
47 {
48 return rpp::observer<Type, observer_strategy>{m_state.lock()};
49 }
50
51 template<rpp::constraint::observer_of_type<Type> TObs>
52 void on_subscribe(TObs&& observer) const
53 {
54 m_state.lock()->on_subscribe(std::forward<TObs>(observer));
55 }
56
57 rpp::disposable_wrapper get_disposable() const
58 {
59 return m_state;
60 }
61
62 private:
63 disposable_wrapper_impl<subject_state<Type>> m_state = disposable_wrapper_impl<subject_state<Type>>::make();
64 };
65} // namespace rpp::subjects::details
