51struct from_iterable_schedulable
52{
53 template<constraint::decayed_type PackedContainer, constraint::observer_strategy<utils::iterable_value_t<PackedContainer>> Strategy>
54 rpp::schedulers::optional_duration operator()(const observer<utils::iterable_value_t<PackedContainer>, Strategy>& obs, const PackedContainer& cont, size_t& index) const
55 {
56 try
57 {
58 auto itr = std::cbegin(cont);
59 auto end = std::cend(cont);
60 std::advance(itr, static_cast<int64_t>(index));
61
62 if (itr != end)
63 {
64 obs.on_next(utils::as_const(*itr));
65 if (std::next(itr) != end) // it was not last
66 {
67 ++index;
68 return schedulers::duration{}; // re-schedule this
69 }
70 }
71
72 obs.on_completed();
73 }
74 catch (...)
75 {
76 obs.on_error(std::current_exception());
77 }
78 return std::nullopt;
79 }
80};

82template<constraint::decayed_type PackedContainer, schedulers::constraint::scheduler TScheduler>
83struct from_iterable_strategy
84{
85public:
86
87 using value_type = rpp::utils::iterable_value_t<PackedContainer>;
88 using expected_disposable_strategy = std::conditional_t<rpp::schedulers::utils::get_worker_t<TScheduler>::is_none_disposable, rpp::details::observables::bool_disposable_strategy_selector, rpp::details::observables::fixed_disposable_strategy_selector<1>>;
89
90 template<typename... Args>
91 from_iterable_strategy(const TScheduler& scheduler, Args&&... args)
92 : container{std::forward<Args>(args)...}
93 , scheduler{scheduler}
94 {
95 }
96
97
98 RPP_NO_UNIQUE_ADDRESS PackedContainer container;
99 RPP_NO_UNIQUE_ADDRESS TScheduler scheduler;

101 template<constraint::observer_strategy<utils::iterable_value_t<PackedContainer>> Strategy>
102 void subscribe(observer<utils::iterable_value_t<PackedContainer>, Strategy>&& obs) const
103 {
104 if constexpr (std::same_as<TScheduler, schedulers::immediate>)
105 {
106 try
107 {
108 for (const auto& v : container)
109 {
110 if (obs.is_disposed())
111 return;
112
113 obs.on_next(v);
114 }
115
116 obs.on_completed();
117 }
118 catch (...)
119 {
120 obs.on_error(std::current_exception());
121 }
122 }
123 else
124 {
125 const auto worker = scheduler.create_worker();
126 if constexpr (!rpp::schedulers::utils::get_worker_t<TScheduler>::is_none_disposable)
127 {
128 if (auto d = worker.get_disposable(); !d.is_disposed())
129 obs.set_upstream(std::move(d));
130 }
131 worker.schedule(from_iterable_schedulable{}, std::move(obs), container, size_t{});
132 }
133 }
134};
