44 template<rpp::constraint::observer TObserver, typename TObservable>
45 struct retry_observer_strategy
46 {
47 using preferred_disposable_strategy = rpp::details::observers::none_disposable_strategy;
48
49 std::shared_ptr<retry_state_t<TObserver, TObservable>> state;
50 mutable bool locally_disposed{};
51
52 template<typename T>
53 void on_next(T&& v) const
54 {
55 state->observer.on_next(std::forward<T>(v));
56 }
57
58 void on_error(const std::exception_ptr& err) const
59 {
60 locally_disposed = true;
61 if (state->count == 0)
62 {
63 state->observer.on_error(err);
64 return;
65 }
66
67 state->disposable.clear();
68
69 if (state->is_inside_drain.exchange(false, std::memory_order::seq_cst))
70 return;
71
72 drain(state);
73 }
74
75 void on_completed() const
76 {
77 locally_disposed = true;
78 state->observer.on_completed();
79 }
80
81 void set_upstream(const disposable_wrapper& d) const
82 {
83 state->disposable.add(d);
84 }
85
86 bool is_disposed() const { return locally_disposed || state->disposable.is_disposed(); }
87 };

89 template<rpp::constraint::observer TObserver, typename TObservable>
90 void drain(const std::shared_ptr<retry_state_t<TObserver, TObservable>>& state)
91 {
92 while (!state->disposable.is_disposed())
93 {
94 if (state->count)
95 --state->count.value();
96 state->is_inside_drain.store(true, std::memory_order::seq_cst);
97 try
98 {
99 using value_type = rpp::utils::extract_observer_type_t<TObserver>;
100 state->observable.subscribe(observer<value_type, retry_observer_strategy<TObserver, TObservable>>{state});
101
102 if (state->is_inside_drain.exchange(false, std::memory_order::seq_cst))
103 return;
104 }
105 catch (...)
106 {
107 state->observer.on_error(std::current_exception());
108 return;
109 }
110 }
