47 template<typename T>
48 void on_next(T&& v) const
49 {
50 // mutex need to be locked during changing of values, generating new values and sending of new values due to we can't update value while we are sending old one
51 const auto observer = state->get_observer_under_lock();
52 state->get_values().template get<I>().emplace(std::forward<T>(v));
53
54 state->get_values().apply(&apply_impl<decltype(state)>, state, observer);
55 }
56
57 private:
58 template<typename TState>
59 static void apply_impl(const TState& disposable, const rpp::utils::pointer_under_lock<Observer>& observer, const std::optional<Args>&... vals)
60 {
61 if ((vals.has_value() && ...))
62 observer->on_next(disposable->get_selector()(vals.value()...));
63 }
64 };
