48 template<typename TDisposable>
49 struct combining_observer_strategy
50 {
51 std::shared_ptr<TDisposable> disposable{};
52
53 void set_upstream(const rpp::disposable_wrapper& d) const
54 {
55 disposable->add(d);
56 }
57
58 bool is_disposed() const
59 {
60 return disposable->is_disposed();
61 }
62
63 void on_error(const std::exception_ptr& err) const
64 {
65 disposable->get_observer_under_lock()->on_error(err);
66 disposable->dispose();
67 }
68
69 void on_completed() const
70 {
71 if (disposable->decrement_on_completed())
72 {
73 disposable->get_observer_under_lock()->on_completed();
74 disposable->dispose();
75 }
76 }
77 };

99 auto lift(Observer&& observer) const

105 template<rpp::constraint::decayed_type Type, rpp::constraint::observer Observer>
106 static auto subscribe_impl(Observer&& observer, const TSelector& selector, const TObservables&... observables)
107 {
108 using Disposable = TDisposable<Observer, TSelector, Type, rpp::utils::extract_observable_type_t<TObservables>...>;
109
110 const auto disposable = disposable_wrapper_impl<Disposable>::make(std::forward<Observer>(observer), selector);
111 auto locked = disposable.lock();
112 locked->get_observer_under_lock()->set_upstream(disposable.as_weak());
113 subscribe<std::decay_t<Type>>(locked, std::index_sequence_for<TObservables...>{}, observables...);
114
115 return rpp::observer<Type, TStrategy<0, std::decay_t<Observer>, TSelector, Type, rpp::utils::extract_observable_type_t<TObservables>...>>{std::move(locked)};
116 }
