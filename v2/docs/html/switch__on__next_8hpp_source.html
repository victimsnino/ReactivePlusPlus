85 template<rpp::constraint::observer TObserver>
86 class switch_on_next_observer_strategy
87 {
88 public:
89 using preferred_disposable_strategy = rpp::details::observers::none_disposable_strategy;
90
91 switch_on_next_observer_strategy(TObserver&& obs)
92 : m_state{init_state(std::move(obs))}
93 {
94 }
95
96 switch_on_next_observer_strategy(const switch_on_next_observer_strategy&) = delete;
97 switch_on_next_observer_strategy(switch_on_next_observer_strategy&&) noexcept = default;
98
99 template<typename T>
100 void on_next(T&& v) const
101 {
102 m_last_refcount.dispose();
103 m_last_refcount = m_state->add_ref();
104 std::forward<T>(v).subscribe(switch_on_next_inner_observer_strategy<TObserver>{m_state, m_last_refcount});
105 }
106
107 void on_error(const std::exception_ptr& err) const
108 {
109 m_state->get_observer()->on_error(err);
110 m_state->dispose();
111 }
112
113 void on_completed() const
114 {
115 m_this_refcount.dispose();
116 if (m_state->is_disposed())
117 m_state->get_observer()->on_completed();
118 }
119
120 void set_upstream(const disposable_wrapper& d) const { m_this_refcount.add(d); }
121 bool is_disposed() const { return m_this_refcount.is_disposed(); }
122
123 private:
124 static std::shared_ptr<switch_on_next_state_t<TObserver>> init_state(TObserver&& observer)
125 {
126 const auto d = disposable_wrapper_impl<switch_on_next_state_t<TObserver>>::make(std::move(observer));
127 auto ptr = d.lock();
128 ptr->get_observer()->set_upstream(d.as_weak());
129 return ptr;
130 }
131
132 private:
133 std::shared_ptr<switch_on_next_state_t<TObserver>> m_state;
134 rpp::composite_disposable_wrapper m_this_refcount = m_state->add_ref();
135 mutable rpp::composite_disposable_wrapper m_last_refcount = composite_disposable_wrapper::empty();
136 };
