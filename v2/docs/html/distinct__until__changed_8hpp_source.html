21template<rpp::constraint::decayed_type Type, rpp::constraint::decayed_type EqualityFn>
22struct distinct_until_changed_observer_strategy
23{
24 using DisposableStrategyToUseWithThis = rpp::details::none_disposable_strategy;
25
26 RPP_NO_UNIQUE_ADDRESS EqualityFn comparator;
27 mutable std::optional<Type> last_value{};
28
29 template<typename T>
30 void on_next(const rpp::constraint::observer auto& obs, T&& v) const
31 {
32 if (last_value.has_value() && comparator(utils::as_const(last_value.value()), rpp::utils::as_const(v)))
33 return;
34
35 last_value.emplace(std::forward<T>(v));
36 obs.on_next(utils::as_const(last_value.value()));
37 }
38
39 constexpr static forwarding_on_error_strategy on_error{};
40 constexpr static forwarding_on_completed_strategy on_completed{};
41 constexpr static forwarding_set_upstream_strategy set_upstream{};
42 constexpr static forwarding_is_disposed_strategy is_disposed{};
43 constexpr static empty_on_subscribe on_subscribe{};
44};
45
46template<rpp::constraint::decayed_type EqualityFn>
47struct distinct_until_changed_t : public operators::details::template_operator_observable_strategy<distinct_until_changed_observer_strategy, EqualityFn>
48{
49 template<rpp::constraint::decayed_type T>
50 requires rpp::constraint::invocable_r_v<bool, EqualityFn, T, T>
51 using ResultValue = T;
52};
53}
