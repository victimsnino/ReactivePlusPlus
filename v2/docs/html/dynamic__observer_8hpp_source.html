37 template<constraint::decayed_type Type>
38 class dynamic_strategy final
39 {
40 public:
41  template<constraint::observer_strategy<Type> Strategy>
42  requires (std::is_rvalue_reference_v<Strategy&&> && !constraint::decayed_same_as<Strategy, dynamic_strategy<Type>>)
43  explicit dynamic_strategy(Strategy&& strategy) // NOLINT(bugprone-forwarding-reference-overload)
44  : m_forwarder{std::make_shared<Strategy>(std::forward<Strategy>(strategy))}
45  , m_vtable{vtable::template create<Strategy>()} {}
46 
47  dynamic_strategy(const dynamic_strategy&) = default;
48  dynamic_strategy(dynamic_strategy&&) noexcept = default;
49 
50  void set_upstream(const composite_disposable& d) noexcept { m_vtable->set_upstream(m_forwarder.get(), d); }
51  bool is_disposed() const noexcept { return m_vtable->is_disposed(m_forwarder.get()); }
52 
53  void on_next(const Type& v) const noexcept { m_vtable->on_next_lvalue(m_forwarder.get(), v); }
54  void on_next(Type&& v) const noexcept { m_vtable->on_next_rvalue(m_forwarder.get(), std::move(v)); }
55  void on_error(const std::exception_ptr& err) const noexcept { m_vtable->on_error(m_forwarder.get(), err); }
56  void on_completed() const noexcept { m_vtable->on_completed(m_forwarder.get()); }
57 
58 private:
59  struct vtable
60  {
61  void (*on_next_lvalue)(const void*, const Type&){};
62  void (*on_next_rvalue)(const void*, Type&&){};
63  void (*on_error)(const void*, const std::exception_ptr&){};
64  void (*on_completed)(const void*){};
65 
66  void (*set_upstream)(void*, const composite_disposable&){};
67  bool (*is_disposed)(const void*){};
68 
69  template<constraint::observer_strategy<Type> Strategy>
70  static const vtable* create() noexcept
71  {
72  static vtable s_res{
73  .on_next_lvalue = forwarding_on_next_lvalue<Type, Strategy>,
74  .on_next_rvalue = forwarding_on_next_rvalue<Type, Strategy>,
75  .on_error = forwarding_on_error<Strategy>,
76  .on_completed = forwarding_on_completed<Strategy>,
77  .set_upstream = forwarding_set_upstream<Strategy>,
78  .is_disposed = forwarding_is_disposed<Strategy>,
79  };
80  return &s_res;
81  }
82  };
