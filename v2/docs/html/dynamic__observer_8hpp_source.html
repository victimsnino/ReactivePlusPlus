20template<auto Fn, bool NoExcept>
21struct member_ptr_caller_impl;
22
23template<bool NoExcept, class T, class R, class... Args, R (T::*F)(Args...) noexcept(NoExcept)>
24struct member_ptr_caller_impl<F, NoExcept>
25{
26 static R call(void* data, Args... args) noexcept(NoExcept) { return (static_cast<T*>(data)->*F)(static_cast<Args>(args)...); }
27};
28
29template<bool NoExcept, class T, class R, class... Args, R (T::*F)(Args...) const noexcept(NoExcept)>
30struct member_ptr_caller_impl<F, NoExcept>
31{
32 static R call(const void* data, Args... args) noexcept(NoExcept) { return (static_cast<const T*>(data)->*F)(static_cast<Args>(args)...); }
33};

35template<auto Fn>
36using member_ptr_caller = member_ptr_caller_impl<Fn, noexcept(Fn)>;

39template<rpp::constraint::decayed_type Type>
40class dynamic_strategy final
41{
42public:
43 template<rpp::constraint::observer_strategy<Type> Strategy>
44 requires (!rpp::constraint::decayed_same_as<Strategy, dynamic_strategy<Type>>)
45 explicit dynamic_strategy(observer<Type, Strategy>&& obs)
46 : m_forwarder{std::make_shared<observer<Type, Strategy>>(std::move(obs))}
47 , m_vtable{vtable::template create<observer<Type, Strategy>>()}
48 {
49 }
50
51 dynamic_strategy(const dynamic_strategy&) = default;
52 dynamic_strategy(dynamic_strategy&&) noexcept = default;
53
54 void set_upstream(const disposable_wrapper& d) noexcept { m_vtable->set_upstream(m_forwarder.get(), d); }
55
56 bool is_disposed() const noexcept { return m_vtable->is_disposed(m_forwarder.get()); }
57
58 void on_next(const Type& v) const noexcept { m_vtable->on_next_lvalue(m_forwarder.get(), v); }
59
60 void on_next(Type&& v) const noexcept { m_vtable->on_next_rvalue(m_forwarder.get(), std::move(v)); }
61
62 void on_error(const std::exception_ptr& err) const noexcept { m_vtable->on_error(m_forwarder.get(), err); }
63
64 void on_completed() const noexcept { m_vtable->on_completed(m_forwarder.get()); }
65
66private:
67 struct vtable
68 {
69 void (*on_next_lvalue)(const void*, const Type&){};
70 void (*on_next_rvalue)(const void*, Type&&){};
71 void (*on_error)(const void*, const std::exception_ptr&){};
72 void (*on_completed)(const void*){};
73
74 void (*set_upstream)(void*, const disposable_wrapper&){};
75 bool (*is_disposed)(const void*){};
76
77 template<rpp::constraint::observer Strategy>
78 static const vtable* create() noexcept
79 {
80 static vtable s_res{
81 .on_next_lvalue = &member_ptr_caller<static_cast<typename Strategy::on_next_lvalue>(&Strategy::on_next)>::call,
82 .on_next_rvalue = &member_ptr_caller<static_cast<typename Strategy::on_next_rvalue>(&Strategy::on_next)>::call,
83 .on_error = &member_ptr_caller<&Strategy::on_error>::call,
84 .on_completed = &member_ptr_caller<&Strategy::on_completed>::call,
85 .set_upstream = &member_ptr_caller<&Strategy::set_upstream>::call,
86 .is_disposed = &member_ptr_caller<&Strategy::is_disposed>::call,
87 };
88 return &s_res;
89 }
90 };
91
92private:
93 std::shared_ptr<void> m_forwarder;
94 const vtable* m_vtable;
95};
