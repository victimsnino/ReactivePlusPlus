26namespace rpp::subjects::details
27{
28struct completed{};
29struct disposed{};
30
31template<rpp::constraint::decayed_type Type>
32class subject_state final : public std::enable_shared_from_this<subject_state<Type>>, public composite_disposable
33{
34 using shared_observers = std::shared_ptr<std::vector<rpp::dynamic_observer<Type>>>;
35 using state_t = std::variant<shared_observers, std::exception_ptr, completed, disposed>;
36
37public:
38 template<rpp::constraint::observer_of_type<Type> TObs>
39 void on_subscribe(TObs&& observer)
40 {
41 std::unique_lock lock{m_mutex};
42 process_state_unsafe(
43 m_state,
44 [&](const shared_observers& observers)
45 {
46 auto new_observers = make_copy_of_subscribed_observers(true, observers);
47 auto observer_as_dynamic = std::forward<TObs>(observer).as_dynamic();
48 new_observers->push_back(observer_as_dynamic);
49 m_state = std::move(new_observers);
50
51 lock.unlock();
52 set_upstream(observer_as_dynamic);
53 },
54 [&](const std::exception_ptr& err)
55 {
56 lock.unlock();
57 observer.on_error(err);
58 },
59 [&](completed)
60 {
61 lock.unlock();
62 observer.on_completed();
63 });
64 }
65
66 void on_next(const Type& v)
67 {
68 if (const auto observers = extract_observers_under_lock_if_there())
69 rpp::utils::for_each(*observers, [&](const auto& sub) { sub.on_next(v); });
70 }
71
72 void on_error(const std::exception_ptr& err)
73 {
74 if (const auto observers = exchange_observers_under_lock_if_there(err))
75 rpp::utils::for_each(*observers, [&](const auto& sub) { sub.on_error(err); });
76 }
77
78 void on_completed()
79 {
80 if (const auto observers = exchange_observers_under_lock_if_there(completed{}))
81 rpp::utils::for_each(*observers, rpp::utils::static_mem_fn<&dynamic_observer<Type>::on_completed>{});
82 }
83
84private:
85 void dispose_impl() override
86 {
87 exchange_observers_under_lock_if_there(disposed{});
88 }
89
90 void set_upstream(rpp::dynamic_observer<Type>& obs)
91 {
92 obs.set_upstream(rpp::disposable_wrapper{make_callback_disposable(
93 [weak = this->weak_from_this()]()
94 {
95 if (const auto shared = weak.lock())
96 {
97 std::unique_lock lock{shared->m_mutex};
98 process_state_unsafe(shared->m_state,
99 [&](const shared_observers& observers)
100 {
101 shared->m_state = make_copy_of_subscribed_observers(false, observers);
102 });
103 }
104 })});
105 }
106
107 static shared_observers make_copy_of_subscribed_observers(bool add, const shared_observers& current_subs)
108 {
109 auto subs = std::make_shared<std::vector<dynamic_observer<Type>>>();
110 subs->reserve(deduce_new_size(add, current_subs));
111 if (current_subs)
112 {
113 std::copy_if(current_subs->cbegin(),
114 current_subs->cend(),
115 std::back_inserter(*subs),
116 rpp::utils::static_not_mem_fn<&dynamic_observer<Type>::is_disposed>{});
117 }
118 return subs;
119 }
120
121 static size_t deduce_new_size(bool add, const shared_observers& current_subs)
122 {
123 if (!current_subs)
124 return add ? 1 : 0;
125
126 if (add)
127 return current_subs->size() + 1;
128
129 return std::max(current_subs->size(), size_t{1}) - 1;
130 }
131
132 static void process_state_unsafe(const state_t& state, const auto&... actions)
133 {
134 std::visit(rpp::utils::overloaded{actions..., rpp::utils::empty_function_any_t{}}, state);
135 }
136
137 shared_observers extract_observers_under_lock_if_there()
138 {
139 std::lock_guard lock{m_mutex};
140
141 if (!std::holds_alternative<shared_observers>(m_state))
142 return {};
143
144 return std::get<shared_observers>(m_state);
145 }
146
147 shared_observers exchange_observers_under_lock_if_there(state_t&& new_val)
148 {
149 std::lock_guard lock{m_mutex};
150
151 if (!std::holds_alternative<shared_observers>(m_state))
152 return {};
153
154 return std::get<shared_observers>(std::exchange(m_state, std::move(new_val)));
155 }
156
157private:
158 state_t m_state{};
159 std::mutex m_mutex{};
160};
