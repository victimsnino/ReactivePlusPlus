29template<typename SubscribeStrategy, rpp::constraint::decayed_type... Args>
30class operator_observable_strategy_base
31{
32public:
33 template<rpp::constraint::decayed_same_as<Args>... TArgs>
34 operator_observable_strategy_base(TArgs&&... args)
35 : m_vals{std::forward<TArgs>(args)...}
36 {
37 }
38
39 template<rpp::constraint::decayed_type Type, rpp::constraint::observer Observer>
40 auto lift(Observer&& observer) const
41 {
42 return m_vals.apply(&SubscribeStrategy::template apply<Type, Observer, Args...>, std::forward<Observer>(observer));
43 }
44
45private:
46 RPP_NO_UNIQUE_ADDRESS rpp::utils::tuple<Args...> m_vals{};
47};
