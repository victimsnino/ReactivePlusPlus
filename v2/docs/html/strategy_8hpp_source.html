27template<typename S, typename Type>
28concept operator_strategy = requires(const S& const_strategy,
29 S& strategy,
30 const Type& v,
31 const composite_disposable disposable,
32 const dynamic_observer<Type>& const_observer,
33 dynamic_observer<Type>& observer)
34{
35 const_strategy.on_next(const_observer, v);
36 const_strategy.on_next(const_observer, Type{});
37 const_strategy.on_error(const_observer, std::exception_ptr{});
38 const_strategy.on_completed(const_observer);
39
40 strategy.set_upstream(observer, disposable);
41 { strategy.is_disposed() } -> std::same_as<bool>;
42};
43}

49template<rpp::constraint::decayed_type T, rpp::constraint::observer_strategy<T> ObserverStrategy, constraint::operator_strategy<T> Strategy>
50class operator_strategy_base<rpp::base_observer<T, ObserverStrategy>, Strategy>
51{
52public:
53 using observer = base_observer<T, ObserverStrategy>;
54
59 template<typename...Args>
60 operator_strategy_base(observer&& observer, Args&&...args)
61 : m_observer{std::in_place_index<0>, observer}
62 , m_strategy{std::forward<Args>(args)...} {}
63
64 operator_strategy_base(const operator_strategy_base&) = delete;
65 operator_strategy_base(operator_strategy_base&& other) noexcept
66 : m_observer{std::in_place_index<1>, std::move(other).get_observer()} {}
67
68 operator_strategy_base& operator=(const operator_strategy_base&) = delete;
69 operator_strategy_base& operator=(operator_strategy_base&&) = delete;
70
71 ~operator_strategy_base() noexcept = default;
72
73 void set_upstream(const composite_disposable& d) { m_strategy.set_upstream(get_observer(), d); }
74 bool is_disposed() const noexcept { return m_strategy.is_disposed() || get_observer().is_disposed(); }
75
76 void on_next(const T& v) const { m_strategy.on_next(get_observer(), v); }
77 void on_next(T&& v) const { m_strategy.on_next(get_observer(), std::move(v)); }
78 void on_error(const std::exception_ptr& err) const { m_strategy.on_error(get_observer(), err); }
79 void on_completed() const { m_strategy.on_completed(get_observer()); }
80
81private:
82 observer&& get_observer() &&
83 {
84 return std::visit([](observer& v) -> observer&& {return std::move(v);}, m_observer);
85 }
86
87 observer& get_observer() &
88 {
89 return std::visit([](observer& v) -> observer& {return v;}, m_observer);
90 }
91
92 const observer& get_observer() const &
93 {
94 return std::visit([](const observer& v) -> const observer& {return v;}, m_observer);
95 }
96
97private:
98 std::variant<std::reference_wrapper<observer>, observer> m_observer;
99 RPP_NO_UNIQUE_ADDRESS Strategy m_strategy;
100};
