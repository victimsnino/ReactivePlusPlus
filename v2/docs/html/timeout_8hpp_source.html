105 struct timeout_t
106 {
107 template<rpp::constraint::decayed_type T>
108 struct operator_traits
109 {
110 static_assert(rpp::constraint::observable_of_type<TFallbackObservable, T>, "TFallbackObservable should be the same type as T");
111
112 using result_type = T;
113
114 constexpr static bool own_current_queue = true;
115 };
116
117 rpp::schedulers::duration period;
118 RPP_NO_UNIQUE_ADDRESS TFallbackObservable fallback;
119 RPP_NO_UNIQUE_ADDRESS TScheduler scheduler;
120
121 template<rpp::constraint::decayed_type Type, rpp::details::observables::constraint::disposable_strategy DisposableStrategy, rpp::constraint::observer Observer>
122 auto lift_with_disposable_strategy(Observer&& observer) const
123 {
124 using worker_t = rpp::schedulers::utils::get_worker_t<TScheduler>;
125 using container = typename DisposableStrategy::disposable_container;
126
127 const auto timeout = worker_t::now() + period;
128
129 const auto disposable = disposable_wrapper_impl<timeout_disposable<std::decay_t<Observer>, TFallbackObservable, container>>::make(std::forward<Observer>(observer), period, fallback, timeout);
130 auto ptr = disposable.lock();
131 ptr->get_observer_with_timeout_under_lock()->observer.set_upstream(disposable.as_weak());
132
133 const auto worker = scheduler.create_worker();
134 using wrapper = timeout_disposable_wrapper<std::decay_t<Observer>, TFallbackObservable, container>;
135 worker.schedule(
136 timeout,
137 [](wrapper& handler) -> rpp::schedulers::optional_delay_to {
138 auto locked_obs_with_timeout = handler.disposable->get_observer_with_timeout_under_lock();
139 if (rpp::schedulers::utils::get_worker_t<TScheduler>::now() < locked_obs_with_timeout->timeout)
140 return rpp::schedulers::delay_to(locked_obs_with_timeout->timeout);
141
142 if (!handler.disposable->is_disposed())
143 {
144 handler.disposable->dispose();
145 handler.disposable->get_fallback().subscribe(std::move(locked_obs_with_timeout->observer));
146 }
147 return std::nullopt;
148 },
149 wrapper{ptr});
150
151 return rpp::observer<Type, timeout_observer_strategy<std::decay_t<Observer>, TFallbackObservable, container, TScheduler>>{std::move(ptr)};
152 }
153 };
