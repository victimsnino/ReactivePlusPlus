91 {
92 const auto obs_with_timeout = disposable->get_observer_with_timeout_under_lock();
93 if (disposable->is_disposed())

114 template<rpp::constraint::decayed_type T>
115 struct operator_traits
116 {
117 static_assert(rpp::constraint::observable_of_type<TFallbackObservable, T>, "TFallbackObservable should be the same type as T");

124 rpp::schedulers::duration period;
125 RPP_NO_UNIQUE_ADDRESS TFallbackObservable fallback;
126 RPP_NO_UNIQUE_ADDRESS TScheduler scheduler;

128 template<rpp::constraint::decayed_type Type, rpp::details::observables::constraint::disposable_strategy DisposableStrategy, rpp::constraint::observer Observer>
129 auto lift_with_disposable_strategy(Observer&& observer) const
130 {
131 using worker_t = rpp::schedulers::utils::get_worker_t<TScheduler>;
132 using container = typename DisposableStrategy::template add<worker_t::is_none_disposable ? 0 : 1>::disposable_container;
133
134 const auto timeout = rpp::schedulers::utils::get_worker_t<TScheduler>::now() + period;
135
136 const auto disposable = disposable_wrapper_impl<timeout_disposable<std::decay_t<Observer>, TFallbackObservable, container>>::make(std::forward<Observer>(observer), period, fallback, timeout);
137 auto ptr = disposable.lock();
138 ptr->get_observer_with_timeout_under_lock()->observer.set_upstream(disposable.as_weak());
139
140 const auto worker = scheduler.create_worker();
141 if constexpr (!rpp::schedulers::utils::get_worker_t<TScheduler>::is_none_disposable)
142 {
143 if (auto d = worker.get_disposable(); !d.is_disposed())
144 disposable.add(std::move(d));
145 }
146
147 using wrapper = timeout_disposable_wrapper<std::decay_t<Observer>, TFallbackObservable, container>;
148 worker.schedule(
149 timeout,
150 [](wrapper& handler) -> rpp::schedulers::optional_delay_to {
151 auto locked_obs_with_timeout = handler.disposable->get_observer_with_timeout_under_lock();
152 if (rpp::schedulers::utils::get_worker_t<TScheduler>::now() < locked_obs_with_timeout->timeout)
153 return rpp::schedulers::delay_to(locked_obs_with_timeout->timeout);
154
155 if (!handler.disposable->is_disposed())
156 {
157 handler.disposable->dispose();
158 handler.disposable->get_fallback().subscribe(std::move(locked_obs_with_timeout->observer));
159 }
160 return std::nullopt;
161 },
162 wrapper{ptr});
163
164 return rpp::observer<Type, timeout_observer_strategy<std::decay_t<Observer>, TFallbackObservable, container, TScheduler>>{std::move(ptr)};
165 }
166 };
