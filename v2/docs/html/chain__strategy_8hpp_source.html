34 observable_chain_strategy(const TStrategy& strategy, const observable_chain_strategy<TStrategies...>& strategies)
35 : m_strategy(strategy)
36 , m_strategies(strategies)
37 {
38 }
39
40 template<rpp::constraint::observer_of_type<value_type> Observer>
41 void subscribe(Observer&& observer) const
42 {
43 if constexpr (rpp::constraint::operator_lift_with_disposable<TStrategy, typename base::value_type, typename base::expected_disposable_strategy>)
44 m_strategies.subscribe(m_strategy.template lift_with_disposable<typename base::value_type, typename base::expected_disposable_strategy>(std::forward<Observer>(observer)));
45 else if constexpr (rpp::constraint::operator_lift<TStrategy, typename base::value_type>)
46 m_strategies.subscribe(m_strategy.template lift<typename base::value_type>(std::forward<Observer>(observer)));
47 else
48 m_strategy.subscribe(std::forward<Observer>(observer), m_strategies);
49 }
50
51private:
52 RPP_NO_UNIQUE_ADDRESS TStrategy m_strategy;
53 RPP_NO_UNIQUE_ADDRESS observable_chain_strategy<TStrategies...> m_strategies;
54};
