43 template<rpp::constraint::observer_of_type<value_type> Observer>
44 void subscribe(Observer&& observer) const
45 {
46 [[maybe_unused]] const auto drain_on_exit = own_current_thread_if_needed();
47
48 if constexpr (rpp::constraint::operator_lift_with_disposable_strategy<TStrategy, typename base::value_type, typename base::expected_disposable_strategy>)
49 m_strategies.subscribe(m_strategy.template lift_with_disposable_strategy<typename base::value_type, typename base::expected_disposable_strategy>(std::forward<Observer>(observer)));
50 else if constexpr (rpp::constraint::operator_lift<TStrategy, typename base::value_type>)
51 m_strategies.subscribe(m_strategy.template lift<typename base::value_type>(std::forward<Observer>(observer)));
52 else
53 m_strategy.subscribe(std::forward<Observer>(observer), m_strategies);
54 }
55
56 private:
57 static auto own_current_thread_if_needed()
58 {
59 if constexpr (requires { requires operator_traits::own_current_queue; })
60 return rpp::schedulers::current_thread::own_queue_and_drain_finally_if_not_owned();
61 else
62 return rpp::utils::none{};
63 }
64
65 private:
66 RPP_NO_UNIQUE_ADDRESS TStrategy m_strategy;
67 RPP_NO_UNIQUE_ADDRESS chain<TStrategies...> m_strategies;
68 };
